77/// How many blocks we import votes from per leaf update.
78///
79/// Since vote import is relatively slow, we have to limit the maximum amount of work we do on leaf
80/// updates (and especially on startup) so the dispute coordinator won't be considered stalling.
81const CHAIN_IMPORT_MAX_BATCH_SIZE: usize = 8;
82
83// Initial data for `dispute-coordinator`. It is provided only at first start.
84pub struct InitialData {
85	pub participations: Vec<(ParticipationPriority, ParticipationRequest)>,
86	pub votes: Vec<ScrapedOnChainVotes>,
87	pub leaf: ActivatedLeaf,
88}

90/// After the first active leaves update we transition to `Initialized` state.
91///
92/// Before the first active leaves update we can't really do much. We cannot check incoming
93/// statements for validity, we cannot query orderings, we have no valid `SessionInfo`,
94/// ...
95pub(crate) struct Initialized {
96	keystore: Arc<LocalKeystore>,
97	runtime_info: RuntimeInfo,
98	/// We have the onchain state of disabled validators as well as the offchain
99	/// state that is based on the lost disputes.
100	offchain_disabled_validators: OffchainDisabledValidators,
101	/// The indices of the controlled validators, cached by session.
102	controlled_validator_indices: ControlledValidatorIndices,
103	/// This is the highest `SessionIndex` seen via `ActiveLeavesUpdate`. It doesn't matter if it
104	/// was cached successfully or not. It is used to detect ancient disputes.
105	highest_session_seen: SessionIndex,
106	/// Will be set to `true` if an error occurred during the last caching attempt
107	gaps_in_cache: bool,
108	spam_slots: SpamSlots,
109	participation: Participation,
110	scraper: ChainScraper,
111	participation_receiver: WorkerMessageReceiver,
112	/// Backlog of still to be imported votes from chain.
113	///
114	/// For some reason importing votes is relatively slow, if there is a large finality lag (~50
115	/// blocks) we will be too slow importing all votes from unfinalized chains on startup
116	/// (dispute-coordinator gets killed because of unresponsiveness).
117	///
118	/// https://github.com/paritytech/polkadot/issues/6912
119	///
120	/// To resolve this, we limit the amount of votes imported at once to
121	/// `CHAIN_IMPORT_MAX_BATCH_SIZE` and put the rest here for later processing.
122	chain_import_backlog: VecDeque<ScrapedOnChainVotes>,
123	metrics: Metrics,
124}

160	/// Run the initialized subsystem.
161	///
162	/// `initial_data` is optional. It is passed on first start and is `None` on subsystem restarts.
163	pub async fn run<B, Context>(
164		mut self,
165		mut ctx: Context,
166		mut backend: B,
167		mut initial_data: Option<InitialData>,
168		clock: Box<dyn Clock>,
169	) -> FatalResult<()>
170	where
171		B: Backend,
172	{
173		loop {
174			let res =
175				self.run_until_error(&mut ctx, &mut backend, &mut initial_data, &*clock).await;
176			if let Ok(()) = res {
177				gum::info!(target: LOG_TARGET, "received `Conclude` signal, exiting");
178				return Ok(())
179			}
180			log_error(res)?;
181		}
182	}
183
184	// Run the subsystem until an error is encountered or a `conclude` signal is received.
185	// Most errors are non-fatal and should lead to another call to this function.
186	//
187	// A return value of `Ok` indicates that an exit should be made, while non-fatal errors
188	// lead to another call to this function.
189	async fn run_until_error<B, Context>(
190		&mut self,
191		ctx: &mut Context,
192		backend: &mut B,
193		initial_data: &mut Option<InitialData>,
194		clock: &dyn Clock,
195	) -> Result<()>
196	where
197		B: Backend,
198	{
199		if let Some(InitialData { participations, votes: on_chain_votes, leaf: first_leaf }) =
200			initial_data.take()
201		{
202			for (priority, request) in participations {
203				self.participation.queue_participation(ctx, priority, request).await?;
204			}
205
206			let mut overlay_db = OverlayedBackend::new(backend);
207
208			self.process_chain_import_backlog(
209				ctx,
210				&mut overlay_db,
211				on_chain_votes,
212				clock.now(),
213				first_leaf.hash,
214			)
215			.await;
216
217			if !overlay_db.is_empty() {
218				let ops = overlay_db.into_write_ops();
219				backend.write(ops)?;
220			}
221
222			// Also provide first leaf to participation for good measure.
223			self.participation
224				.process_active_leaves_update(ctx, &ActiveLeavesUpdate::start_work(first_leaf))
225				.await?;
226		}
227
228		loop {
229			gum::trace!(target: LOG_TARGET, "Waiting for message");
230			let mut overlay_db = OverlayedBackend::new(backend);
231			let default_confirm = Box::new(|| Ok(()));
232			let confirm_write =
233				match MuxedMessage::receive(ctx, &mut self.participation_receiver).await? {
234					MuxedMessage::Participation(msg) => {
235						gum::trace!(target: LOG_TARGET, "MuxedMessage::Participation");
236						let ParticipationStatement {
237							session,
238							candidate_hash,
239							candidate_receipt,
240							outcome,
241						} = self.participation.get_participation_result(ctx, msg).await?;
242						if let Some(valid) = outcome.validity() {
243							gum::trace!(
244								target: LOG_TARGET,
245								?session,
246								?candidate_hash,
247								?valid,
248								"Issuing local statement based on participation outcome."
249							);
250							self.issue_local_statement(
251								ctx,
252								&mut overlay_db,
253								candidate_hash,
254								candidate_receipt,
255								session,
256								valid,
257								clock.now(),
258							)
259							.await?;
260						} else {
261							gum::warn!(target: LOG_TARGET, ?outcome, "Dispute participation failed");
262						}
263						default_confirm
264					},
265					MuxedMessage::Subsystem(msg) => match msg {
266						FromOrchestra::Signal(OverseerSignal::Conclude) => return Ok(()),
267						FromOrchestra::Signal(OverseerSignal::ActiveLeaves(update)) => {
268							gum::trace!(target: LOG_TARGET, "OverseerSignal::ActiveLeaves");
269							self.process_active_leaves_update(
270								ctx,
271								&mut overlay_db,
272								update,
273								clock.now(),
274							)
275							.await?;
276							default_confirm
277						},
278						FromOrchestra::Signal(OverseerSignal::BlockFinalized(_, n)) => {
279							gum::trace!(target: LOG_TARGET, "OverseerSignal::BlockFinalized");
280							self.scraper.process_finalized_block(&n);
281							default_confirm
282						},
283						FromOrchestra::Communication { msg } =>
284							self.handle_incoming(ctx, &mut overlay_db, msg, clock.now()).await?,
285					},
286				};
287
288			if !overlay_db.is_empty() {
289				let ops = overlay_db.into_write_ops();
290				backend.write(ops)?;
291			}
292			// even if the changeset was empty,
293			// otherwise the caller will error.
294			confirm_write()?;
295		}
296	}

298	async fn process_active_leaves_update<Context>(
299		&mut self,
300		ctx: &mut Context,
301		overlay_db: &mut OverlayedBackend<'_, impl Backend>,
302		update: ActiveLeavesUpdate,
303		now: u64,
304	) -> Result<()> {
305		gum::trace!(target: LOG_TARGET, timestamp = now, "Processing ActiveLeavesUpdate");
306		let scraped_updates =
307			self.scraper.process_active_leaves_update(ctx.sender(), &update).await?;
308		log_error(
309			self.participation
310				.bump_to_priority_for_candidates(ctx, &scraped_updates.included_receipts)
311				.await,
312		)?;
313		self.participation.process_active_leaves_update(ctx, &update).await?;
314
315		if let Some(new_leaf) = update.activated {
316			gum::trace!(
317				target: LOG_TARGET,
318				leaf_hash = ?new_leaf.hash,
319				block_number = new_leaf.number,
320				"Processing ActivatedLeaf"
321			);
322
323			let session_idx =
324				self.runtime_info.get_session_index_for_child(ctx.sender(), new_leaf.hash).await;
325
326			match session_idx {
327				Ok(session_idx)
328					if self.gaps_in_cache || session_idx > self.highest_session_seen =>
329				{
330					// Pin the block from the new session.
331					self.runtime_info.pin_block(session_idx, new_leaf.unpin_handle);
332					// Fetch the last `DISPUTE_WINDOW` number of sessions unless there are no gaps
333					// in cache and we are not missing too many `SessionInfo`s
334					let prune_up_to = session_idx.saturating_sub(DISPUTE_WINDOW.get() - 1);
335					let fetch_lower_bound =
336						if !self.gaps_in_cache && self.highest_session_seen > prune_up_to {
337							self.highest_session_seen + 1
338						} else {
339							prune_up_to
340						};
341
342					// There is a new session. Perform a dummy fetch to cache it.
343					for idx in fetch_lower_bound..=session_idx {
344						if let Err(err) = self
345							.runtime_info
346							.get_session_info_by_index(ctx.sender(), new_leaf.hash, idx)
347							.await
348						{
349							gum::debug!(
350								target: LOG_TARGET,
351								session_idx,
352								leaf_hash = ?new_leaf.hash,
353								?err,
354								"Error caching SessionInfo on ActiveLeaves update"
355							);
356							self.gaps_in_cache = true;
357						}
358					}
359
360					self.highest_session_seen = session_idx;
361
362					db::v1::note_earliest_session(overlay_db, prune_up_to)?;
363					self.spam_slots.prune_old(prune_up_to);
364					self.offchain_disabled_validators.prune_old(prune_up_to);
365				},
366				Ok(_) => { /* no new session => nothing to cache */ },
367				Err(err) => {
368					gum::debug!(
369						target: LOG_TARGET,
370						?err,
371						"Failed to update session cache for disputes - can't fetch session index",
372					);
373				},
374			}
375
376			let ScrapedUpdates { unapplied_slashes, on_chain_votes, .. } = scraped_updates;
377
378			self.process_unapplied_slashes(ctx, new_leaf.hash, unapplied_slashes).await;
379
380			gum::trace!(
381				target: LOG_TARGET,
382				timestamp = now,
383				"Will process {} onchain votes",
384				on_chain_votes.len()
385			);
386
387			self.process_chain_import_backlog(ctx, overlay_db, on_chain_votes, now, new_leaf.hash)
388				.await;
389		}
390
391		gum::trace!(target: LOG_TARGET, timestamp = now, "Done processing ActiveLeavesUpdate");
392		Ok(())
393	}

395	/// For each unapplied (past-session) slash, report an unsigned extrinsic
396	/// to the runtime.
397	async fn process_unapplied_slashes<Context>(
398		&mut self,
399		ctx: &mut Context,
400		relay_parent: Hash,
401		unapplied_slashes: Vec<(SessionIndex, CandidateHash, slashing::PendingSlashes)>,
402	) {
403		for (session_index, candidate_hash, pending) in unapplied_slashes {
404			gum::info!(
405				target: LOG_TARGET,
406				?session_index,
407				?candidate_hash,
408				n_slashes = pending.keys.len(),
409				"Processing unapplied validator slashes",
410			);
411
412			let pinned_hash = self.runtime_info.get_block_in_session(session_index);
413			let inclusions = self.scraper.get_blocks_including_candidate(&candidate_hash);
414			if pinned_hash.is_none() && inclusions.is_empty() {
415				gum::info!(
416					target: LOG_TARGET,
417					?session_index,
418					"Couldn't find blocks in the session for an unapplied slash",
419				);
420				return
421			}
422
423			// Find a relay block that we can use
424			// to generate key ownership proof on.
425			// We use inclusion parents as a fallback.
426			let mut key_ownership_proofs = Vec::new();
427			let mut dispute_proofs = Vec::new();
428
429			let blocks_in_the_session =
430				pinned_hash.into_iter().chain(inclusions.into_iter().map(|(_n, h)| h));
431			for hash in blocks_in_the_session {
432				for (validator_index, validator_id) in pending.keys.iter() {
433					let res = key_ownership_proof(ctx.sender(), hash, validator_id.clone()).await;
434
435					match res {
436						Ok(Some(key_ownership_proof)) => {
437							key_ownership_proofs.push(key_ownership_proof);
438							let time_slot =
439								slashing::DisputesTimeSlot::new(session_index, candidate_hash);
440							let dispute_proof = slashing::DisputeProof {
441								time_slot,
442								kind: pending.kind,
443								validator_index: *validator_index,
444								validator_id: validator_id.clone(),
445							};
446							dispute_proofs.push(dispute_proof);
447						},
448						Ok(None) => {},
449						Err(runtime::Error::RuntimeRequest(RuntimeApiError::NotSupported {
450							..
451						})) => {
452							gum::debug!(
453								target: LOG_TARGET,
454								?session_index,
455								?candidate_hash,
456								?validator_id,
457								"Key ownership proof not yet supported.",
458							);
459						},
460						Err(error) => {
461							gum::warn!(
462								target: LOG_TARGET,
463								?error,
464								?session_index,
465								?candidate_hash,
466								?validator_id,
467								"Could not generate key ownership proof",
468							);
469						},
470					}
471				}
472
473				if !key_ownership_proofs.is_empty() {
474					// If we found a parent that we can use, stop searching.
475					// If one key ownership was resolved successfully, all of them should be.
476					debug_assert_eq!(key_ownership_proofs.len(), pending.keys.len());
477					break
478				}
479			}
480
481			let expected_keys = pending.keys.len();
482			let resolved_keys = key_ownership_proofs.len();
483			if resolved_keys < expected_keys {
484				gum::warn!(
485					target: LOG_TARGET,
486					?session_index,
487					?candidate_hash,
488					"Could not generate key ownership proofs for {} keys",
489					expected_keys - resolved_keys,
490				);
491			}
492			debug_assert_eq!(resolved_keys, dispute_proofs.len());
493
494			for (key_ownership_proof, dispute_proof) in
495				key_ownership_proofs.into_iter().zip(dispute_proofs.into_iter())
496			{
497				let validator_id = dispute_proof.validator_id.clone();
498
499				gum::info!(
500					target: LOG_TARGET,
501					?session_index,
502					?candidate_hash,
503					key_ownership_proof_len = key_ownership_proof.len(),
504					"Trying to submit a slashing report",
505				);
506
507				let res = submit_report_dispute_lost(
508					ctx.sender(),
509					relay_parent,
510					dispute_proof,
511					key_ownership_proof,
512				)
513				.await;
514
515				match res {
516					Err(runtime::Error::RuntimeRequest(RuntimeApiError::NotSupported {
517						..
518					})) => {
519						gum::debug!(
520							target: LOG_TARGET,
521							?session_index,
522							?candidate_hash,
523							"Reporting pending slash not yet supported",
524						);
525					},
526					Err(error) => {
527						gum::warn!(
528							target: LOG_TARGET,
529							?error,
530							?session_index,
531							?candidate_hash,
532							"Error reporting pending slash",
533						);
534					},
535					Ok(Some(())) => {
536						gum::info!(
537							target: LOG_TARGET,
538							?session_index,
539							?candidate_hash,
540							?validator_id,
541							"Successfully reported pending slash",
542						);
543					},
544					Ok(None) => {
545						gum::debug!(
546							target: LOG_TARGET,
547							?session_index,
548							?candidate_hash,
549							?validator_id,
550							"Duplicate pending slash report",
551						);
552					},
553				}
554			}
555		}
556	}

558	/// Process one batch of our `chain_import_backlog`.
559	///
560	/// `new_votes` will be appended beforehand.
561	async fn process_chain_import_backlog<Context>(
562		&mut self,
563		ctx: &mut Context,
564		overlay_db: &mut OverlayedBackend<'_, impl Backend>,
565		new_votes: Vec<ScrapedOnChainVotes>,
566		now: u64,
567		block_hash: Hash,
568	) {
569		let mut chain_import_backlog = std::mem::take(&mut self.chain_import_backlog);
570		chain_import_backlog.extend(new_votes);
571		let import_range =
572			0..std::cmp::min(CHAIN_IMPORT_MAX_BATCH_SIZE, chain_import_backlog.len());
573		// The `runtime-api` subsystem has an internal queue which serializes the execution,
574		// so there is no point in running these in parallel
575		for votes in chain_import_backlog.drain(import_range) {
576			let res = self.process_on_chain_votes(ctx, overlay_db, votes, now, block_hash).await;
577			match res {
578				Ok(()) => {},
579				Err(error) => {
580					gum::warn!(target: LOG_TARGET, ?error, "Skipping scraping block due to error",);
581				},
582			};
583		}
584		self.chain_import_backlog = chain_import_backlog;
585	}

587	/// Scrapes on-chain votes (backing votes and concluded disputes) for a active leaf of the
588	/// relay chain.
589	async fn process_on_chain_votes<Context>(
590		&mut self,
591		ctx: &mut Context,
592		overlay_db: &mut OverlayedBackend<'_, impl Backend>,
593		votes: ScrapedOnChainVotes,
594		now: u64,
595		block_hash: Hash,
596	) -> Result<()> {
597		let ScrapedOnChainVotes { session, backing_validators_per_candidate, disputes } = votes;
598
599		if backing_validators_per_candidate.is_empty() && disputes.is_empty() {
600			return Ok(())
601		}
602
603		// Scraped on-chain backing votes for the candidates with
604		// the new active leaf as if we received them via gossip.
605		for (candidate_receipt, backers) in backing_validators_per_candidate {
606			// Obtain the session info, for sake of `ValidatorId`s
607			let relay_parent = candidate_receipt.descriptor.relay_parent();
608			let session_info = match self
609				.runtime_info
610				.get_session_info_by_index(ctx.sender(), relay_parent, session)
611				.await
612			{
613				Ok(extended_session_info) => &extended_session_info.session_info,
614				Err(err) => {
615					gum::warn!(
616						target: LOG_TARGET,
617						?session,
618						?err,
619						"Could not retrieve session info from RuntimeInfo",
620					);
621					return Ok(())
622				},
623			};
624
625			let candidate_hash = candidate_receipt.hash();
626			gum::trace!(
627				target: LOG_TARGET,
628				?candidate_hash,
629				?relay_parent,
630				"Importing backing votes from chain for candidate"
631			);
632			let statements = backers
633				.into_iter()
634				.filter_map(|(validator_index, attestation)| {
635					let validator_public: ValidatorId = session_info
636						.validators
637						.get(validator_index)
638						.or_else(|| {
639							gum::error!(
640								target: LOG_TARGET,
641								?session,
642								?validator_index,
643								"Missing public key for validator",
644							);
645							None
646						})
647						.cloned()?;
648					let validator_signature = attestation.signature().clone();
649					let valid_statement_kind =
650						match attestation.to_compact_statement(candidate_hash) {
651							CompactStatement::Seconded(_) =>
652								ValidDisputeStatementKind::BackingSeconded(relay_parent),
653							CompactStatement::Valid(_) =>
654								ValidDisputeStatementKind::BackingValid(relay_parent),
655						};
656					debug_assert!(
657						SignedDisputeStatement::new_checked(
658							DisputeStatement::Valid(valid_statement_kind.clone()),
659							candidate_hash,
660							session,
661							validator_public.clone(),
662							validator_signature.clone(),
663						).is_ok(),
664						"Scraped backing votes had invalid signature! candidate: {:?}, session: {:?}, validator_public: {:?}, validator_index: {}",
665						candidate_hash,
666						session,
667						validator_public,
668						validator_index.0,
669					);
670					let signed_dispute_statement =
671						SignedDisputeStatement::new_unchecked_from_trusted_source(
672							DisputeStatement::Valid(valid_statement_kind.clone()),
673							candidate_hash,
674							session,
675							validator_public,
676							validator_signature,
677						);
678					Some((signed_dispute_statement, validator_index))
679				})
680				.collect();
681
682			// Importantly, handling import statements for backing votes also
683			// clears spam slots for any newly backed candidates
684			let import_result = self
685				.handle_import_statements(
686					ctx,
687					overlay_db,
688					MaybeCandidateReceipt::Provides(candidate_receipt),
689					session,
690					statements,
691					now,
692				)
693				.await?;
694			match import_result {
695				ImportStatementsResult::ValidImport => gum::trace!(
696					target: LOG_TARGET,
697					?relay_parent,
698					?session,
699					"Imported backing votes from chain"
700				),
701				ImportStatementsResult::InvalidImport => gum::warn!(
702					target: LOG_TARGET,
703					?relay_parent,
704					?session,
705					"Attempted import of on-chain backing votes failed"
706				),
707			}
708		}
709
710		// Import disputes from on-chain, this already went through a vote so it's assumed
711		// as verified. This will only be stored, gossiping it is not necessary.
712		for DisputeStatementSet { candidate_hash, session, statements } in disputes {
713			gum::trace!(
714				target: LOG_TARGET,
715				?candidate_hash,
716				?session,
717				"Importing dispute votes from chain for candidate"
718			);
719			let session_info = match self
720				.runtime_info
721				.get_session_info_by_index(ctx.sender(), block_hash, session)
722				.await
723			{
724				Ok(extended_session_info) => &extended_session_info.session_info,
725				Err(err) => {
726					gum::warn!(
727						target: LOG_TARGET,
728						?candidate_hash,
729						?session,
730						?err,
731						"Could not retrieve session info for recently concluded dispute"
732					);
733					continue
734				},
735			};
736
737			let statements = statements
738				.into_iter()
739				.filter_map(|(dispute_statement, validator_index, validator_signature)| {
740					let validator_public: ValidatorId = session_info
741						.validators
742						.get(validator_index)
743						.or_else(|| {
744							gum::error!(
745								target: LOG_TARGET,
746								?candidate_hash,
747								?session,
748								"Missing public key for validator {:?} that participated in concluded dispute",
749								&validator_index
750							);
751							None
752						})
753						.cloned()?;
754
755					Some((
756						SignedDisputeStatement::new_unchecked_from_trusted_source(
757							dispute_statement,
758							candidate_hash,
759							session,
760							validator_public,
761							validator_signature,
762						),
763						validator_index,
764					))
765				})
766				.collect::<Vec<_>>();
767			if statements.is_empty() {
768				gum::debug!(target: LOG_TARGET, "Skipping empty from chain dispute import");
769				continue
770			}
771			let import_result = self
772				.handle_import_statements(
773					ctx,
774					overlay_db,
775					// TODO <https://github.com/paritytech/polkadot/issues/4011>
776					MaybeCandidateReceipt::AssumeBackingVotePresent(candidate_hash),
777					session,
778					statements,
779					now,
780				)
781				.await?;
782			match import_result {
783				ImportStatementsResult::ValidImport => gum::trace!(
784					target: LOG_TARGET,
785					?candidate_hash,
786					?session,
787					"Imported statement of dispute from on-chain"
788				),
789				ImportStatementsResult::InvalidImport => gum::warn!(
790					target: LOG_TARGET,
791					?candidate_hash,
792					?session,
793					"Attempted import of on-chain statement of dispute failed"
794				),
795			}
796		}
797
798		Ok(())
799	}

801	async fn handle_incoming<Context>(
802		&mut self,
803		ctx: &mut Context,
804		overlay_db: &mut OverlayedBackend<'_, impl Backend>,
805		message: DisputeCoordinatorMessage,
806		now: Timestamp,
807	) -> Result<Box<dyn FnOnce() -> JfyiResult<()>>> {
808		match message {
809			DisputeCoordinatorMessage::ImportStatements {
810				candidate_receipt,
811				session,
812				statements,
813				pending_confirmation,
814			} => {
815				gum::trace!(
816					target: LOG_TARGET,
817					candidate_hash = ?candidate_receipt.hash(),
818					?session,
819					"DisputeCoordinatorMessage::ImportStatements"
820				);
821				let outcome = self
822					.handle_import_statements(
823						ctx,
824						overlay_db,
825						MaybeCandidateReceipt::Provides(candidate_receipt),
826						session,
827						statements,
828						now,
829					)
830					.await?;
831				let report = move || match pending_confirmation {
832					Some(pending_confirmation) => pending_confirmation
833						.send(outcome)
834						.map_err(|_| JfyiError::DisputeImportOneshotSend),
835					None => Ok(()),
836				};
837
838				match outcome {
839					ImportStatementsResult::InvalidImport => {
840						report()?;
841					},
842					// In case of valid import, delay confirmation until actual disk write:
843					ImportStatementsResult::ValidImport => return Ok(Box::new(report)),
844				}
845			},
846			DisputeCoordinatorMessage::RecentDisputes(tx) => {
847				gum::trace!(target: LOG_TARGET, "Loading recent disputes from db");
848				let recent_disputes = if let Some(disputes) = overlay_db.load_recent_disputes()? {
849					disputes
850				} else {
851					BTreeMap::new()
852				};
853				gum::trace!(target: LOG_TARGET, "Loaded recent disputes from db");
854
855				let _ = tx.send(
856					recent_disputes.into_iter().map(|(k, v)| (k.0, k.1, v)).collect::<Vec<_>>(),
857				);
858			},
859			DisputeCoordinatorMessage::ActiveDisputes(tx) => {
860				gum::trace!(target: LOG_TARGET, "DisputeCoordinatorMessage::ActiveDisputes");
861				let recent_disputes = if let Some(disputes) = overlay_db.load_recent_disputes()? {
862					disputes
863				} else {
864					BTreeMap::new()
865				};
866
867				let _ = tx.send(
868					get_active_with_status(recent_disputes.into_iter(), now)
869						.map(|((session_idx, candidate_hash), dispute_status)| {
870							(session_idx, candidate_hash, dispute_status)
871						})
872						.collect(),
873				);
874			},
875			DisputeCoordinatorMessage::QueryCandidateVotes(query, tx) => {
876				gum::trace!(target: LOG_TARGET, "DisputeCoordinatorMessage::QueryCandidateVotes");
877				let mut query_output = Vec::new();
878				for (session_index, candidate_hash) in query {
879					if let Some(v) =
880						overlay_db.load_candidate_votes(session_index, &candidate_hash)?
881					{
882						query_output.push((session_index, candidate_hash, v.into()));
883					} else {
884						gum::debug!(
885							target: LOG_TARGET,
886							session_index,
887							"No votes found for candidate",
888						);
889					}
890				}
891				let _ = tx.send(query_output);
892			},
893			DisputeCoordinatorMessage::IssueLocalStatement(
894				session,
895				candidate_hash,
896				candidate_receipt,
897				valid,
898			) => {
899				gum::trace!(target: LOG_TARGET, "DisputeCoordinatorMessage::IssueLocalStatement");
900				self.issue_local_statement(
901					ctx,
902					overlay_db,
903					candidate_hash,
904					candidate_receipt,
905					session,
906					valid,
907					now,
908				)
909				.await?;
910			},
911			DisputeCoordinatorMessage::DetermineUndisputedChain {
912				base: (base_number, base_hash),
913				block_descriptions,
914				tx,
915			} => {
916				gum::trace!(
917					target: LOG_TARGET,
918					"DisputeCoordinatorMessage::DetermineUndisputedChain"
919				);
920
921				let undisputed_chain = determine_undisputed_chain(
922					overlay_db,
923					base_number,
924					base_hash,
925					block_descriptions,
926				)?;
927
928				let _ = tx.send(undisputed_chain);
929			},
930		}
931
932		Ok(Box::new(|| Ok(())))
933	}
934
935	// We use fatal result rather than result here. Reason being, We for example increase
936	// spam slots in this function. If then the import fails for some non fatal and
937	// unrelated reason, we should likely actually decrement previously incremented spam
938	// slots again, for non fatal errors - which is cumbersome and actually not needed
939	async fn handle_import_statements<Context>(
940		&mut self,
941		ctx: &mut Context,
942		overlay_db: &mut OverlayedBackend<'_, impl Backend>,
943		candidate_receipt: MaybeCandidateReceipt,
944		session: SessionIndex,
945		statements: Vec<(SignedDisputeStatement, ValidatorIndex)>,
946		now: Timestamp,
947	) -> FatalResult<ImportStatementsResult> {
948		gum::trace!(target: LOG_TARGET, ?statements, "In handle import statements");
949		if self.session_is_ancient(session) {
950			// It is not valid to participate in an ancient dispute (spam?) or too new.
951			return Ok(ImportStatementsResult::InvalidImport)
952		}
953
954		let candidate_hash = candidate_receipt.hash();
955		let votes_in_db = overlay_db.load_candidate_votes(session, &candidate_hash)?;
956		let relay_parent = match &candidate_receipt {
957			MaybeCandidateReceipt::Provides(candidate_receipt) =>
958				candidate_receipt.descriptor().relay_parent(),
959			MaybeCandidateReceipt::AssumeBackingVotePresent(candidate_hash) => match &votes_in_db {
960				Some(votes) => votes.candidate_receipt.descriptor().relay_parent(),
961				None => {
962					gum::warn!(
963						target: LOG_TARGET,
964						session,
965						?candidate_hash,
966						"Cannot obtain relay parent without `CandidateReceipt` available!"
967					);
968					return Ok(ImportStatementsResult::InvalidImport)
969				},
970			},
971		};
972
973		let env = match CandidateEnvironment::new(
974			ctx,
975			&mut self.runtime_info,
976			session,
977			relay_parent,
978			self.offchain_disabled_validators.iter(session),
979			&mut self.controlled_validator_indices,
980		)
981		.await
982		{
983			None => {
984				gum::warn!(
985					target: LOG_TARGET,
986					session,
987					"We are lacking a `SessionInfo` for handling import of statements."
988				);
989
990				return Ok(ImportStatementsResult::InvalidImport)
991			},
992			Some(env) => env,
993		};
994
995		let n_validators = env.validators().len();
996
997		gum::trace!(
998			target: LOG_TARGET,
999			?candidate_hash,
1000			?session,
1001			?n_validators,
1002			"Number of validators"
1003		);
1004
1005		// In case we are not provided with a candidate receipt
1006		// we operate under the assumption, that a previous vote
1007		// which included a `CandidateReceipt` was seen.
1008		// This holds since every block is preceded by the `Backing`-phase.
1009		//
1010		// There is one exception: A sufficiently sophisticated attacker could prevent
1011		// us from seeing the backing votes by withholding arbitrary blocks, and hence we do
1012		// not have a `CandidateReceipt` available.
1013		let old_state = match votes_in_db.map(CandidateVotes::from) {
1014			Some(votes) => CandidateVoteState::new(votes, &env, now),
1015			None =>
1016				if let MaybeCandidateReceipt::Provides(candidate_receipt) = candidate_receipt {
1017					CandidateVoteState::new_from_receipt(candidate_receipt)
1018				} else {
1019					gum::warn!(
1020						target: LOG_TARGET,
1021						session,
1022						?candidate_hash,
1023						"Cannot import votes, without `CandidateReceipt` available!"
1024					);
1025					return Ok(ImportStatementsResult::InvalidImport)
1026				},
1027		};
1028
1029		gum::trace!(target: LOG_TARGET, ?candidate_hash, ?session, "Loaded votes");
1030
1031		let controlled_indices = env.controlled_indices();
1032		let own_statements = statements
1033			.iter()
1034			.filter(|(statement, validator_index)| {
1035				controlled_indices.contains(validator_index) &&
1036					*statement.candidate_hash() == candidate_hash
1037			})
1038			.cloned()
1039			.collect::<Vec<_>>();
1040
1041		let import_result = {
1042			let intermediate_result = old_state.import_statements(&env, statements, now);
1043
1044			// Handle approval vote import:
1045			//
1046			// See guide: We import on fresh disputes to maximize likelihood of fetching votes for
1047			// dead forks and once concluded to maximize time for approval votes to trickle in.
1048			if intermediate_result.is_freshly_disputed() ||
1049				intermediate_result.is_freshly_concluded()
1050			{
1051				gum::trace!(
1052					target: LOG_TARGET,
1053					?candidate_hash,
1054					?session,
1055					"Requesting approval signatures"
1056				);
1057				let (tx, rx) = oneshot::channel();
1058				// Use of unbounded channels justified because:
1059				// 1. Only triggered twice per dispute.
1060				// 2. Raising a dispute is costly (requires validation + recovery) by honest nodes,
1061				// dishonest nodes are limited by spam slots.
1062				// 3. Concluding a dispute is even more costly.
1063				// Therefore it is reasonable to expect a simple vote request to succeed way faster
1064				// than disputes are raised.
1065				// 4. We are waiting (and blocking the whole subsystem) on a response right after -
1066				// therefore even with all else failing we will never have more than
1067				// one message in flight at any given time.
1068				ctx.send_unbounded_message(
1069					ApprovalVotingParallelMessage::GetApprovalSignaturesForCandidate(
1070						candidate_hash,
1071						tx,
1072					),
1073				);
1074
1075				match rx.await {
1076					Err(_) => {
1077						gum::warn!(
1078							target: LOG_TARGET,
1079							"Fetch for approval votes got cancelled, only expected during shutdown!"
1080						);
1081						intermediate_result
1082					},
1083					Ok(votes) => {
1084						gum::trace!(
1085							target: LOG_TARGET,
1086							count = votes.len(),
1087							"Successfully received approval votes."
1088						);
1089						intermediate_result.import_approval_votes(&env, votes, now)
1090					},
1091				}
1092			} else {
1093				gum::trace!(
1094					target: LOG_TARGET,
1095					?candidate_hash,
1096					?session,
1097					"Not requested approval signatures"
1098				);
1099				intermediate_result
1100			}
1101		};
1102
1103		gum::trace!(
1104			target: LOG_TARGET,
1105			?candidate_hash,
1106			?session,
1107			?n_validators,
1108			"Import result ready"
1109		);
1110
1111		let new_state = import_result.new_state();
1112
1113		let is_included = self.scraper.is_candidate_included(&candidate_hash);
1114		let is_backed = self.scraper.is_candidate_backed(&candidate_hash);
1115		let own_vote_missing = new_state.own_vote_missing();
1116		let is_disputed = new_state.is_disputed();
1117		let is_confirmed = new_state.is_confirmed();
1118		let is_disabled = |v: &ValidatorIndex| env.disabled_indices().contains(v);
1119		let potential_spam =
1120			is_potential_spam(&self.scraper, &new_state, &candidate_hash, is_disabled);
1121		let allow_participation = !potential_spam;
1122
1123		gum::trace!(
1124			target: LOG_TARGET,
1125			?own_vote_missing,
1126			?potential_spam,
1127			?is_included,
1128			?candidate_hash,
1129			confirmed = ?new_state.is_confirmed(),
1130			has_invalid_voters = ?!import_result.new_invalid_voters().is_empty(),
1131			n_disabled_validators = ?env.disabled_indices().len(),
1132			"Is spam?"
1133		);
1134
1135		// This check is responsible for all clearing of spam slots. It runs
1136		// whenever a vote is imported from on or off chain, and decrements
1137		// slots whenever a candidate is newly backed, confirmed, or has our
1138		// own vote.
1139		if !potential_spam {
1140			self.spam_slots.clear(&(session, candidate_hash));
1141
1142		// Potential spam:
1143		} else if !import_result.new_invalid_voters().is_empty() {
1144			let mut free_spam_slots_available = false;
1145			// Only allow import if at least one validator voting invalid, has not exceeded
1146			// its spam slots:
1147			for index in import_result.new_invalid_voters() {
1148				// Disputes can only be triggered via an invalidity stating vote, thus we only
1149				// need to increase spam slots on invalid votes. (If we did not, we would also
1150				// increase spam slots for backing validators for example - as validators have to
1151				// provide some opposing vote for dispute-distribution).
1152				free_spam_slots_available |=
1153					self.spam_slots.add_unconfirmed(session, candidate_hash, *index);
1154			}
1155			if !free_spam_slots_available {
1156				gum::debug!(
1157					target: LOG_TARGET,
1158					?candidate_hash,
1159					?session,
1160					invalid_voters = ?import_result.new_invalid_voters(),
1161					"Rejecting import because of full spam slots."
1162				);
1163				return Ok(ImportStatementsResult::InvalidImport)
1164			}
1165		}
1166
1167		// Participate in dispute if we did not cast a vote before and actually have keys to cast a
1168		// local vote. Disputes should fall in one of the categories below, otherwise we will
1169		// refrain from participation:
1170		// - `is_included` lands in prioritised queue
1171		// - `is_confirmed` | `is_backed` lands in best effort queue
1172		// We don't participate in disputes on finalized candidates.
1173		if own_vote_missing && is_disputed && allow_participation {
1174			let priority = ParticipationPriority::with_priority_if(is_included);
1175			gum::trace!(
1176				target: LOG_TARGET,
1177				?candidate_hash,
1178				?priority,
1179				"Queuing participation for candidate"
1180			);
1181			if priority.is_priority() {
1182				self.metrics.on_queued_priority_participation();
1183			} else {
1184				self.metrics.on_queued_best_effort_participation();
1185			}
1186			let request_timer = self.metrics.time_participation_pipeline();
1187			let r = self
1188				.participation
1189				.queue_participation(
1190					ctx,
1191					priority,
1192					ParticipationRequest::new(
1193						new_state.candidate_receipt().clone(),
1194						session,
1195						env.executor_params().clone(),
1196						request_timer,
1197					),
1198				)
1199				.await;
1200			log_error(r)?;
1201		} else {
1202			gum::trace!(
1203				target: LOG_TARGET,
1204				?candidate_hash,
1205				?is_confirmed,
1206				?own_vote_missing,
1207				?is_disputed,
1208				?allow_participation,
1209				?is_included,
1210				?is_backed,
1211				"Will not queue participation for candidate"
1212			);
1213
1214			if !allow_participation {
1215				self.metrics.on_refrained_participation();
1216			}
1217		}
1218
1219		// Also send any already existing approval vote on new disputes:
1220		if import_result.is_freshly_disputed() {
1221			let our_approval_votes = new_state.own_approval_votes().into_iter().flatten();
1222			for (validator_index, sig) in our_approval_votes {
1223				let pub_key = match env.validators().get(validator_index) {
1224					None => {
1225						gum::error!(
1226							target: LOG_TARGET,
1227							?validator_index,
1228							?session,
1229							"Could not find pub key in `SessionInfo` for our own approval vote!"
1230						);
1231						continue
1232					},
1233					Some(k) => k,
1234				};
1235				let statement = SignedDisputeStatement::new_unchecked_from_trusted_source(
1236					DisputeStatement::Valid(ValidDisputeStatementKind::ApprovalChecking),
1237					candidate_hash,
1238					session,
1239					pub_key.clone(),
1240					sig.clone(),
1241				);
1242				gum::trace!(
1243					target: LOG_TARGET,
1244					?candidate_hash,
1245					?session,
1246					?validator_index,
1247					"Sending out own approval vote"
1248				);
1249				match make_dispute_message(
1250					env.session_info(),
1251					&new_state.votes(),
1252					statement,
1253					validator_index,
1254				) {
1255					Err(err) => {
1256						gum::error!(
1257							target: LOG_TARGET,
1258							?err,
1259							"No ongoing dispute, but we checked there is one!"
1260						);
1261					},
1262					Ok(dispute_message) => {
1263						ctx.send_message(DisputeDistributionMessage::SendDispute(dispute_message))
1264							.await;
1265					},
1266				};
1267			}
1268		}
1269
1270		// All good, update recent disputes if state has changed:
1271		if let Some(new_status) = new_state.dispute_status() {
1272			// Only bother with db access, if there was an actual change.
1273			if import_result.dispute_state_changed() {
1274				let mut recent_disputes = overlay_db.load_recent_disputes()?.unwrap_or_default();
1275
1276				let status =
1277					recent_disputes.entry((session, candidate_hash)).or_insert_with(|| {
1278						gum::info!(
1279							target: LOG_TARGET,
1280							?candidate_hash,
1281							session,
1282							"New dispute initiated for candidate.",
1283						);
1284						DisputeStatus::active()
1285					});
1286
1287				*status = *new_status;
1288
1289				gum::trace!(
1290					target: LOG_TARGET,
1291					?candidate_hash,
1292					?status,
1293					has_concluded_for = ?new_state.has_concluded_for(),
1294					has_concluded_against = ?new_state.has_concluded_against(),
1295					"Writing recent disputes with updates for candidate"
1296				);
1297				overlay_db.write_recent_disputes(recent_disputes);
1298			}
1299		}
1300
1301		// Notify ChainSelection if a dispute has concluded against a candidate. ChainSelection
1302		// will need to mark the candidate's relay parent as reverted.
1303		if import_result.has_fresh_byzantine_threshold_against() {
1304			let blocks_including = self.scraper.get_blocks_including_candidate(&candidate_hash);
1305			for (parent_block_number, parent_block_hash) in &blocks_including {
1306				gum::warn!(
1307					target: LOG_TARGET,
1308					?candidate_hash,
1309					?parent_block_number,
1310					?parent_block_hash,
1311					"Dispute has just concluded against the candidate hash noted. Its parent will be marked as reverted."
1312				);
1313			}
1314			if blocks_including.len() > 0 {
1315				ctx.send_message(ChainSelectionMessage::RevertBlocks(blocks_including)).await;
1316			} else {
1317				gum::warn!(
1318					target: LOG_TARGET,
1319					?candidate_hash,
1320					?session,
1321					"Could not find an including block for candidate against which a dispute has concluded."
1322				);
1323			}
1324		}
1325
1326		// Update metrics:
1327		if import_result.is_freshly_disputed() {
1328			self.metrics.on_open();
1329		}
1330		self.metrics.on_valid_votes(import_result.imported_valid_votes());
1331		self.metrics.on_invalid_votes(import_result.imported_invalid_votes());
1332		gum::trace!(
1333			target: LOG_TARGET,
1334			?candidate_hash,
1335			?session,
1336			imported_approval_votes = ?import_result.imported_approval_votes(),
1337			imported_valid_votes = ?import_result.imported_valid_votes(),
1338			imported_invalid_votes = ?import_result.imported_invalid_votes(),
1339			total_valid_votes = ?import_result.new_state().votes().valid.raw().len(),
1340			total_invalid_votes = ?import_result.new_state().votes().invalid.len(),
1341			confirmed = ?import_result.new_state().is_confirmed(),
1342			"Import summary"
1343		);
1344
1345		self.metrics.on_approval_votes(import_result.imported_approval_votes());
1346		if import_result.is_freshly_concluded_for() {
1347			gum::info!(
1348				target: LOG_TARGET,
1349				?candidate_hash,
1350				session,
1351				"Dispute on candidate concluded with 'valid' result",
1352			);
1353			for (statement, validator_index) in own_statements.iter() {
1354				if statement.statement().indicates_invalidity() {
1355					gum::warn!(
1356						target: LOG_TARGET,
1357						?candidate_hash,
1358						?validator_index,
1359						"Voted against a candidate that was concluded valid.",
1360					);
1361				}
1362			}
1363			for validator_index in new_state.votes().invalid.keys() {
1364				gum::info!(
1365					target: LOG_TARGET,
1366					?candidate_hash,
1367					?validator_index,
1368					?session,
1369					"Disabled offchain for voting invalid against a valid candidate",
1370				);
1371				self.offchain_disabled_validators
1372					.insert_against_valid(session, *validator_index);
1373			}
1374			self.metrics.on_concluded_valid();
1375		}
1376		if import_result.is_freshly_concluded_against() {
1377			gum::info!(
1378				target: LOG_TARGET,
1379				?candidate_hash,
1380				session,
1381				"Dispute on candidate concluded with 'invalid' result",
1382			);
1383			for (statement, validator_index) in own_statements.iter() {
1384				if statement.statement().indicates_validity() {
1385					gum::warn!(
1386						target: LOG_TARGET,
1387						?candidate_hash,
1388						?validator_index,
1389						"Voted for a candidate that was concluded invalid.",
1390					);
1391				}
1392			}
1393			for (validator_index, (kind, _sig)) in new_state.votes().valid.raw() {
1394				let is_backer = kind.is_backing();
1395				gum::info!(
1396					target: LOG_TARGET,
1397					?candidate_hash,
1398					?validator_index,
1399					?session,
1400					?is_backer,
1401					"Disabled offchain for voting valid for an invalid candidate",
1402				);
1403				self.offchain_disabled_validators.insert_for_invalid(
1404					session,
1405					*validator_index,
1406					is_backer,
1407				);
1408			}
1409			self.metrics.on_concluded_invalid();
1410		}
1411
1412		// After validators are disabled, revisit active disputes to unactivate those where all
1413		// raising parties are now disabled
1414		if import_result.is_freshly_concluded_for() || import_result.is_freshly_concluded_against()
1415		{
1416			self.revisit_active_disputes_after_disabling(overlay_db, session)?;
1417		}
1418
1419		// Only write when votes have changed.
1420		if let Some(votes) = import_result.into_updated_votes() {
1421			overlay_db.write_candidate_votes(session, candidate_hash, votes.into());
1422		}
1423
1424		Ok(ImportStatementsResult::ValidImport)
1425	}

1427	async fn issue_local_statement<Context>(
1428		&mut self,
1429		ctx: &mut Context,
1430		overlay_db: &mut OverlayedBackend<'_, impl Backend>,
1431		candidate_hash: CandidateHash,
1432		candidate_receipt: CandidateReceipt,
1433		session: SessionIndex,
1434		valid: bool,
1435		now: Timestamp,
1436	) -> Result<()> {
1437		gum::trace!(
1438			target: LOG_TARGET,
1439			?candidate_hash,
1440			?session,
1441			?valid,
1442			?now,
1443			"Issuing local statement for candidate!"
1444		);
1445
1446		// Load environment:
1447		let env = match CandidateEnvironment::new(
1448			ctx,
1449			&mut self.runtime_info,
1450			session,
1451			candidate_receipt.descriptor.relay_parent(),
1452			self.offchain_disabled_validators.iter(session),
1453			&mut self.controlled_validator_indices,
1454		)
1455		.await
1456		{
1457			None => {
1458				gum::warn!(
1459					target: LOG_TARGET,
1460					session,
1461					"Missing info for session which has an active dispute",
1462				);
1463
1464				return Ok(())
1465			},
1466			Some(env) => env,
1467		};
1468
1469		let votes = overlay_db
1470			.load_candidate_votes(session, &candidate_hash)?
1471			.map(CandidateVotes::from)
1472			.unwrap_or_else(|| CandidateVotes {
1473				candidate_receipt: candidate_receipt.clone(),
1474				valid: ValidCandidateVotes::new(),
1475				invalid: BTreeMap::new(),
1476			});
1477
1478		// Sign a statement for each validator index we control which has
1479		// not already voted. This should generally be maximum 1 statement.
1480		let voted_indices = votes.voted_indices();
1481		let mut statements = Vec::new();
1482
1483		let controlled_indices = env.controlled_indices();
1484		for index in controlled_indices {
1485			if voted_indices.contains(&index) {
1486				continue
1487			}
1488
1489			let keystore = self.keystore.clone() as Arc<_>;
1490			let res = SignedDisputeStatement::sign_explicit(
1491				&keystore,
1492				valid,
1493				candidate_hash,
1494				session,
1495				env.validators()
1496					.get(*index)
1497					.expect("`controlled_indices` are derived from `validators`; qed")
1498					.clone(),
1499			);
1500
1501			match res {
1502				Ok(Some(signed_dispute_statement)) => {
1503					statements.push((signed_dispute_statement, *index));
1504				},
1505				Ok(None) => {},
1506				Err(err) => {
1507					gum::error!(
1508						target: LOG_TARGET,
1509						?err,
1510						"Encountered keystore error while signing dispute statement",
1511					);
1512				},
1513			}
1514		}
1515
1516		// Get our message out:
1517		for (statement, index) in &statements {
1518			let dispute_message =
1519				match make_dispute_message(env.session_info(), &votes, statement.clone(), *index) {
1520					Err(err) => {
1521						gum::debug!(target: LOG_TARGET, ?err, "Creating dispute message failed.");
1522						continue
1523					},
1524					Ok(dispute_message) => dispute_message,
1525				};
1526
1527			ctx.send_message(DisputeDistributionMessage::SendDispute(dispute_message)).await;
1528		}
1529
1530		// Do import
1531		if !statements.is_empty() {
1532			match self
1533				.handle_import_statements(
1534					ctx,
1535					overlay_db,
1536					MaybeCandidateReceipt::Provides(candidate_receipt),
1537					session,
1538					statements,
1539					now,
1540				)
1541				.await?
1542			{
1543				ImportStatementsResult::InvalidImport => {
1544					gum::error!(
1545						target: LOG_TARGET,
1546						?candidate_hash,
1547						?session,
1548						"`handle_import_statements` considers our own votes invalid!"
1549					);
1550				},
1551				ImportStatementsResult::ValidImport => {
1552					gum::trace!(
1553						target: LOG_TARGET,
1554						?candidate_hash,
1555						?session,
1556						"`handle_import_statements` successfully imported our vote!"
1557					);
1558				},
1559			}
1560		}
1561
1562		Ok(())
1563	}

1569	/// Revisit active non-confirmed disputes after validators have been disabled.
1570	/// Unactivates disputes where all raising parties (invalid voters) are now disabled.
1571	fn revisit_active_disputes_after_disabling(
1572		&mut self,
1573		overlay_db: &mut OverlayedBackend<'_, impl Backend>,
1574		session: SessionIndex,
1575	) -> FatalResult<()> {
1576		let mut recent_disputes = overlay_db.load_recent_disputes()?.unwrap_or_default();
1577		let mut disputes_to_remove = Vec::new();
1578
1579		// Create session bounds for efficient iteration
1580		let session_start = (session, CandidateHash(Hash::zero()));
1581		let session_end = (session + 1, CandidateHash(Hash::zero()));
1582
1583		for ((dispute_session, candidate_hash), status) in
1584			recent_disputes.range(&session_start..&session_end)
1585		{
1586			debug_assert_eq!(session, *dispute_session);
1587			// Only check unconfirmed
1588			if status.is_confirmed_concluded() {
1589				continue
1590			}
1591			let Some(votes) = overlay_db.load_candidate_votes(*dispute_session, candidate_hash)?
1592			else {
1593				continue
1594			};
1595			// Check if all invalid voters (raising parties) are disabled
1596			if !votes.invalid.is_empty() &&
1597				votes.invalid.iter().all(|(_, validator_index, _)| {
1598					self.offchain_disabled_validators.is_disabled(session, *validator_index)
1599				}) {
1600				disputes_to_remove.push((*dispute_session, *candidate_hash));
1601
1602				gum::info!(
1603					target: LOG_TARGET,
1604					session = dispute_session,
1605					?candidate_hash,
1606					invalid_voters = ?votes.invalid.iter().map(|(_, idx, _)| *idx).collect::<Vec<_>>(),
1607					"Unactivating dispute where all raising parties are now disabled"
1608				);
1609			}
1610		}
1611
1612		// Remove them from RecentDisputes (setting status to inactive)
1613		if !disputes_to_remove.is_empty() {
1614			for key in disputes_to_remove {
1615				recent_disputes.remove(&key);
1616				self.metrics.on_unactivated_dispute();
1617			}
1618			overlay_db.write_recent_disputes(recent_disputes);
1619		}
1620
1621		Ok(())
1622	}
1623}

1625/// Messages to be handled in this subsystem.
1626enum MuxedMessage {
1627	/// Messages from other subsystems.
1628	Subsystem(FromOrchestra<DisputeCoordinatorMessage>),
1629	/// Messages from participation workers.
1630	Participation(participation::WorkerMessage),
1631}
1632
1633#[overseer::contextbounds(DisputeCoordinator, prefix = self::overseer)]
1634impl MuxedMessage {
1635	async fn receive<Context>(
1636		ctx: &mut Context,
1637		from_sender: &mut participation::WorkerMessageReceiver,
1638	) -> FatalResult<Self> {
1639		// We are only fusing here to make `select` happy, in reality we will quit if the stream
1640		// ends.
1641		let from_overseer = ctx.recv().fuse();
1642		futures::pin_mut!(from_overseer, from_sender);
1643		futures::select!(
1644			msg = from_overseer => Ok(Self::Subsystem(msg.map_err(FatalError::SubsystemReceive)?)),
1645			msg = from_sender.next() => Ok(Self::Participation(msg.ok_or(FatalError::ParticipationWorkerReceiverExhausted)?)),
1646		)
1647	}
1648}

1668/// Determine the best block and its block number.
1669/// Assumes `block_descriptions` are sorted from the one
1670/// with the lowest `BlockNumber` to the highest.
1671fn determine_undisputed_chain(
1672	overlay_db: &mut OverlayedBackend<'_, impl Backend>,
1673	base_number: BlockNumber,
1674	base_hash: Hash,
1675	block_descriptions: Vec<BlockDescription>,
1676) -> Result<(BlockNumber, Hash)> {
1677	let last = block_descriptions
1678		.last()
1679		.map(|e| (base_number + block_descriptions.len() as BlockNumber, e.block_hash))
1680		.unwrap_or((base_number, base_hash));
1681
1682	// Fast path for no disputes.
1683	let recent_disputes = match overlay_db.load_recent_disputes()? {
1684		None => return Ok(last),
1685		Some(a) if a.is_empty() => return Ok(last),
1686		Some(a) => a,
1687	};
1688
1689	let is_possibly_invalid = |session, candidate_hash| {
1690		recent_disputes
1691			.get(&(session, candidate_hash))
1692			.map_or(false, |status| status.is_possibly_invalid())
1693	};
1694
1695	for (i, BlockDescription { session, candidates, .. }) in block_descriptions.iter().enumerate() {
1696		if candidates.iter().any(|c| is_possibly_invalid(*session, *c)) {
1697			if i == 0 {
1698				return Ok((base_number, base_hash))
1699			} else {
1700				return Ok((base_number + i as BlockNumber, block_descriptions[i - 1].block_hash))
1701			}
1702		}
1703	}
1704
1705	Ok(last)
1706}

1708/// Ideally, we want to use the top `byzantine_threshold` offenders here based on the amount of
1709/// stake slashed. However, given that slashing might be applied with a delay, we want to have
1710/// some list of offenders as soon as disputes conclude offchain. This list only approximates
1711/// the top offenders and only accounts for lost disputes. But that should be good enough to
1712/// prevent spam attacks.
1713#[derive(Default)]
1714pub struct OffchainDisabledValidators {
1715	per_session: BTreeMap<SessionIndex, LostSessionDisputes>,
1716}
1717
1718struct LostSessionDisputes {
1719	// We separate lost disputes to prioritize "for invalid" offenders. And among those, we
1720	// prioritize backing votes the most. There's no need to limit the size of these sets, as they
1721	// are already limited by the number of validators in the session. We use `LruMap` to ensure
1722	// the iteration order prioritizes most recently disputes lost over older ones in case we reach
1723	// the limit.
1724	backers_for_invalid: LruMap<ValidatorIndex, (), UnlimitedCompact>,
1725	for_invalid: LruMap<ValidatorIndex, (), UnlimitedCompact>,
1726	against_valid: LruMap<ValidatorIndex, (), UnlimitedCompact>,
1727}
1728
1729impl Default for LostSessionDisputes {
1730	fn default() -> Self {
1731		Self {
1732			backers_for_invalid: LruMap::new(UnlimitedCompact),
1733			for_invalid: LruMap::new(UnlimitedCompact),
1734			against_valid: LruMap::new(UnlimitedCompact),
1735		}
1736	}
1737}
1738
1739impl OffchainDisabledValidators {
1740	/// Creates a new instance populated from concluded disputes
1741	pub fn new_from_state(
1742		disputes: &RecentDisputes,
1743		load_candidate_votes: impl Fn(SessionIndex, &CandidateHash) -> Option<CandidateVotes>,
1744		earliest_session: SessionIndex,
1745	) -> Self {
1746		let mut disabled_validators = Self::default();
1747
1748		// Process concluded disputes to identify validators that should be disabled
1749		for ((session, candidate_hash), dispute_status) in disputes {
1750			let session = *session;
1751			// Only process concluded disputes
1752			if dispute_status.concluded_at().is_none() {
1753				continue
1754			}
1755			if session < earliest_session {
1756				continue
1757			}
1758
1759			// Get votes for this dispute
1760			let votes = match load_candidate_votes(session, candidate_hash) {
1761				Some(votes) => votes,
1762				None => continue,
1763			};
1764
1765			// Process votes based on dispute outcome
1766			if dispute_status.has_concluded_for() {
1767				// Dispute concluded with candidate being valid - track validators that voted
1768				// against
1769				for (validator_index, _) in votes.invalid.iter() {
1770					disabled_validators.insert_against_valid(session, *validator_index);
1771				}
1772			} else if dispute_status.has_concluded_against() {
1773				// Dispute concluded with candidate being invalid - track validators that voted for
1774				for (validator_index, (kind, _)) in votes.valid.raw().iter() {
1775					let is_backer = kind.is_backing();
1776					disabled_validators.insert_for_invalid(session, *validator_index, is_backer);
1777				}
1778			}
1779		}
1780
1781		disabled_validators
1782	}

1784	/// Prune state for ancient disputes.
1785	pub fn prune_old(&mut self, up_to_excluding: SessionIndex) {
1786		// split_off returns everything after the given key, including the key.
1787		let mut relevant = self.per_session.split_off(&up_to_excluding);
1788		std::mem::swap(&mut relevant, &mut self.per_session);
1789	}
1790
1791	/// Disable a validator who voted for an invalid candidate.
1792	pub fn insert_for_invalid(
1793		&mut self,
1794		session_index: SessionIndex,
1795		validator_index: ValidatorIndex,
1796		is_backer: bool,
1797	) {
1798		let entry = self.per_session.entry(session_index).or_default();
1799		if is_backer {
1800			entry.backers_for_invalid.insert(validator_index, ());
1801		} else {
1802			entry.for_invalid.insert(validator_index, ());
1803		}
1804	}
1805
1806	/// Disable a validator who voted against a valid candidate.
1807	pub fn insert_against_valid(
1808		&mut self,
1809		session_index: SessionIndex,
1810		validator_index: ValidatorIndex,
1811	) {
1812		self.per_session
1813			.entry(session_index)
1814			.or_default()
1815			.against_valid
1816			.insert(validator_index, ());
1817	}
1818
1819	/// Iterate over all validators that are offchain disabled.
1820	/// The order of iteration prioritizes `for_invalid` offenders (and backers among those) over
1821	/// `against_valid` offenders. And most recently lost disputes over older ones.
1822	/// NOTE: the iterator might contain duplicates.
1823	pub fn iter(&self, session_index: SessionIndex) -> impl Iterator<Item = ValidatorIndex> + '_ {
1824		self.per_session.get(&session_index).into_iter().flat_map(|e| {
1825			e.backers_for_invalid
1826				.iter()
1827				.chain(e.for_invalid.iter())
1828				.chain(e.against_valid.iter())
1829				.map(|(i, _)| *i)
1830		})
1831	}
1832
1833	/// Check if a validator is disabled for a given session.
1834	pub fn is_disabled(
1835		&self,
1836		session_index: SessionIndex,
1837		validator_index: ValidatorIndex,
1838	) -> bool {
1839		self.per_session
1840			.get(&session_index)
1841			.map(|session_disputes| {
1842				session_disputes.backers_for_invalid.peek(&validator_index).is_some() ||
1843					session_disputes.for_invalid.peek(&validator_index).is_some() ||
1844					session_disputes.against_valid.peek(&validator_index).is_some()
1845			})
1846			.unwrap_or(false)
1847	}
