18//! The Cumulus [`Proposer`] is a wrapper around a Substrate [`sp_consensus::Environment`]
19//! for creating new parachain blocks.
20//!
21//! This utility is designed to be composed within any collator consensus algorithm.
22
23use async_trait::async_trait;
24
25use sp_consensus::{EnableProofRecording, Environment, Proposal, Proposer as SubstrateProposer};
26use sp_inherents::{InherentData, InherentDataProvider};
27use sp_runtime::{traits::Block as BlockT, Digest};
28use sp_state_machine::StorageProof;
29
30use cumulus_primitives_parachain_inherent::ParachainInherentData;
31use std::{fmt::Debug, time::Duration};
32
33/// Errors that can occur when proposing a parachain block.
34#[derive(thiserror::Error, Debug)]
35#[error(transparent)]
36pub struct Error {
37	inner: anyhow::Error,
38}
39
40impl Error {
41	/// Create an error tied to the creation of a proposer.
42	pub fn proposer_creation(err: impl Into<anyhow::Error>) -> Self {
43		Error { inner: err.into().context("Proposer Creation") }
44	}
45
46	/// Create an error tied to the proposing logic itself.
47	pub fn proposing(err: impl Into<anyhow::Error>) -> Self {
48		Error { inner: err.into().context("Proposing") }
49	}
50}
51
52/// A type alias for easily referring to the type of a proposal produced by a specific
53/// [`Proposer`].
54pub type ProposalOf<B> = Proposal<B, StorageProof>;
55
56/// An interface for proposers.
57#[async_trait]
58pub trait ProposerInterface<Block: BlockT> {
59	/// Propose a collation using the supplied `InherentData` and the provided
60	/// `ParachainInherentData`.
61	///
62	/// Also specify any required inherent digests, the maximum proposal duration,
63	/// and the block size limit in bytes. See the documentation on
64	/// [`sp_consensus::Proposer::propose`] for more details on how to interpret these parameters.
65	///
66	/// The `InherentData` and `Digest` are left deliberately general in order to accommodate
67	/// all possible collator selection algorithms or inherent creation mechanisms,
68	/// while the `ParachainInherentData` is made explicit so it will be constructed appropriately.
69	///
70	/// If the `InherentData` passed into this function already has a `ParachainInherentData`,
71	/// this should throw an error.
72	async fn propose(
73		&mut self,
74		parent_header: &Block::Header,
75		paras_inherent_data: &ParachainInherentData,
76		other_inherent_data: InherentData,
77		inherent_digests: Digest,
78		max_duration: Duration,
79		block_size_limit: Option<usize>,
80	) -> Result<Option<Proposal<Block, StorageProof>>, Error>;
81}
82
83/// A simple wrapper around a Substrate proposer for creating collations.
84pub struct Proposer<B, T> {
85	inner: T,
86	_marker: std::marker::PhantomData<B>,
87}
88
89impl<B, T> Proposer<B, T> {
90	/// Create a new Cumulus [`Proposer`].
91	pub fn new(inner: T) -> Self {
92		Proposer { inner, _marker: std::marker::PhantomData }
93	}
94}
95
96#[async_trait]
97impl<B, T> ProposerInterface<B> for Proposer<B, T>
98where
99	B: sp_runtime::traits::Block,
100	T: Environment<B> + Send,
101	T::Error: Send + Sync + 'static,
102	T::Proposer: SubstrateProposer<B, ProofRecording = EnableProofRecording, Proof = StorageProof>,
103	<T::Proposer as SubstrateProposer<B>>::Error: Send + Sync + 'static,
104{
105	async fn propose(
106		&mut self,
107		parent_header: &B::Header,
108		paras_inherent_data: &ParachainInherentData,
109		other_inherent_data: InherentData,
110		inherent_digests: Digest,
111		max_duration: Duration,
112		block_size_limit: Option<usize>,
113	) -> Result<Option<Proposal<B, StorageProof>>, Error> {
114		let proposer = self
115			.inner
116			.init(parent_header)
117			.await
118			.map_err(|e| Error::proposer_creation(anyhow::Error::new(e)))?;
119
120		let mut inherent_data = other_inherent_data;
121		paras_inherent_data
122			.provide_inherent_data(&mut inherent_data)
123			.await
124			.map_err(|e| Error::proposing(anyhow::Error::new(e)))?;
125
126		proposer
127			.propose(inherent_data, inherent_digests, max_duration, block_size_limit)
128			.await
129			.map(Some)
130			.map_err(|e| Error::proposing(anyhow::Error::new(e)).into())
131	}
132}
