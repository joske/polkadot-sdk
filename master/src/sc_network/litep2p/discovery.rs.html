80/// Discovery events.
81#[derive(Debug)]
82pub enum DiscoveryEvent {
83	/// Ping RTT measured for peer.
84	Ping {
85		/// Remote peer ID.
86		peer: PeerId,
87
88		/// Ping round-trip time.
89		rtt: Duration,
90	},
91
92	/// Peer identified over `/ipfs/identify/1.0.0` protocol.
93	Identified {
94		/// Peer ID.
95		peer: PeerId,
96
97		/// Listen addresses.
98		listen_addresses: Vec<Multiaddr>,
99
100		/// Supported protocols.
101		supported_protocols: HashSet<ProtocolName>,
102	},
103
104	/// One or more addresses discovered.
105	///
106	/// This event is emitted when a new peer is discovered over mDNS.
107	Discovered {
108		/// Discovered addresses.
109		addresses: Vec<Multiaddr>,
110	},
111
112	/// Routing table has been updated.
113	RoutingTableUpdate {
114		/// Peers that were added to routing table.
115		peers: HashSet<PeerId>,
116	},
117
118	/// New external address discovered.
119	ExternalAddressDiscovered {
120		/// Discovered address.
121		address: Multiaddr,
122	},
123
124	/// The external address has expired.
125	///
126	/// This happens when the internal buffers exceed the maximum number of external addresses,
127	/// and this address is the oldest one.
128	ExternalAddressExpired {
129		/// Expired address.
130		address: Multiaddr,
131	},
132
133	/// `FIND_NODE` query succeeded.
134	FindNodeSuccess {
135		/// Query ID.
136		query_id: QueryId,
137
138		/// Target.
139		target: PeerId,
140
141		/// Found peers.
142		peers: Vec<(PeerId, Vec<Multiaddr>)>,
143	},
144
145	/// `GetRecord` query succeeded.
146	GetRecordSuccess {
147		/// Query ID.
148		query_id: QueryId,
149	},
150
151	/// Record was found from the DHT.
152	GetRecordPartialResult {
153		/// Query ID.
154		query_id: QueryId,
155
156		/// Record.
157		record: PeerRecord,
158	},
159
160	/// Record was successfully stored on the DHT.
161	PutRecordSuccess {
162		/// Query ID.
163		query_id: QueryId,
164	},
165
166	/// Providers were successfully retrieved.
167	GetProvidersSuccess {
168		/// Query ID.
169		query_id: QueryId,
170		/// Found providers sorted by distance to provided key.
171		providers: Vec<ContentProvider>,
172	},
173
174	/// Query failed.
175	QueryFailed {
176		/// Query ID.
177		query_id: QueryId,
178	},
179
180	/// Incoming record to store.
181	IncomingRecord {
182		/// Record.
183		record: Record,
184	},
185
186	/// Started a random Kademlia query.
187	RandomKademliaStarted,
188}

240/// Legacy (fallback) Kademlia protocol name based on `protocol_id`.
241fn legacy_kademlia_protocol_name(id: &ProtocolId) -> ProtocolName {
242	ProtocolName::from(format!("/{}/kad", id.as_ref()))
243}
244
245/// Kademlia protocol name based on `genesis_hash` and `fork_id`.
246fn kademlia_protocol_name<Hash: AsRef<[u8]>>(
247	genesis_hash: Hash,
248	fork_id: Option<&str>,
249) -> ProtocolName {
250	let genesis_hash_hex = bytes2hex("", genesis_hash.as_ref());
251	let protocol = if let Some(fork_id) = fork_id {
252		format!("/{}/{}/kad", genesis_hash_hex, fork_id)
253	} else {
254		format!("/{}/kad", genesis_hash_hex)
255	};
256
257	ProtocolName::from(protocol)
258}

261	/// Create new [`Discovery`].
262	///
263	/// Enables `/ipfs/ping/1.0.0` and `/ipfs/identify/1.0.0` by default and starts
264	/// the mDNS peer discovery if it was enabled.
265	pub fn new<Hash: AsRef<[u8]> + Clone>(
266		local_peer_id: litep2p::PeerId,
267		config: &NetworkConfiguration,
268		genesis_hash: Hash,
269		fork_id: Option<&str>,
270		protocol_id: &ProtocolId,
271		known_peers: HashMap<PeerId, Vec<Multiaddr>>,
272		listen_addresses: Arc<RwLock<HashSet<Multiaddr>>>,
273		_peerstore_handle: Arc<dyn PeerStoreProvider>,
274	) -> (Self, PingConfig, IdentifyConfig, KademliaConfig, Option<MdnsConfig>) {
275		let (ping_config, ping_event_stream) = PingConfig::default();
276		let user_agent = format!("{} ({}) (litep2p)", config.client_version, config.node_name);
277
278		let (identify_config, identify_event_stream) =
279			IdentifyConfig::new("/substrate/1.0".to_string(), Some(user_agent));
280
281		let (mdns_config, mdns_event_stream) = match config.transport {
282			crate::config::TransportConfig::Normal { enable_mdns, .. } => match enable_mdns {
283				true => {
284					let (mdns_config, mdns_event_stream) = MdnsConfig::new(MDNS_QUERY_INTERVAL);
285					(Some(mdns_config), Some(mdns_event_stream))
286				},
287				false => (None, None),
288			},
289			_ => panic!("memory transport not supported"),
290		};
291
292		let (kademlia_config, kademlia_handle) = {
293			let protocol_names = vec![
294				kademlia_protocol_name(genesis_hash.clone(), fork_id),
295				legacy_kademlia_protocol_name(protocol_id),
296			];
297
298			KademliaConfigBuilder::new()
299				.with_known_peers(known_peers)
300				.with_protocol_names(protocol_names)
301				.with_incoming_records_validation_mode(IncomingRecordValidationMode::Manual)
302				.build()
303		};
304
305		(
306			Self {
307				local_peer_id,
308				ping_event_stream,
309				identify_event_stream,
310				mdns_event_stream,
311				kademlia_handle,
312				_peerstore_handle,
313				listen_addresses,
314				random_walk_query_id: None,
315				pending_events: VecDeque::new(),
316				duration_to_next_find_query: Duration::from_secs(1),
317				address_confirmations: LruMap::new(ByLength::new(MAX_EXTERNAL_ADDRESSES)),
318				allow_non_global_addresses: config.allow_non_globals_in_dht,
319				public_addresses: config.public_addresses.iter().cloned().map(Into::into).collect(),
320				next_kad_query: Some(Delay::new(KADEMLIA_QUERY_INTERVAL)),
321				local_protocols: HashSet::from_iter([kademlia_protocol_name(
322					genesis_hash,
323					fork_id,
324				)]),
325			},
326			ping_config,
327			identify_config,
328			kademlia_config,
329			mdns_config,
330		)
331	}

339	/// Add self-reported addresses to routing table if `peer` supports
340	/// at least one of the locally supported DHT protocol.
341	pub async fn add_self_reported_address(
342		&mut self,
343		peer: PeerId,
344		supported_protocols: HashSet<ProtocolName>,
345		addresses: Vec<Multiaddr>,
346	) {
347		if self.local_protocols.is_disjoint(&supported_protocols) {
348			log::trace!(
349				target: LOG_TARGET,
350				"Ignoring self-reported address of peer {peer} as remote node is not part of the \
351				 Kademlia DHT supported by the local node.",
352			);
353			return
354		}
355
356		let addresses = addresses
357			.into_iter()
358			.filter_map(|address| {
359				if !self.allow_non_global_addresses && !Discovery::can_add_to_dht(&address) {
360					log::trace!(
361						target: LOG_TARGET,
362						"ignoring self-reported non-global address {address} from {peer}."
363					);
364
365					return None
366				}
367
368				Some(address)
369			})
370			.collect();
371
372		log::trace!(
373			target: LOG_TARGET,
374			"add self-reported addresses for {peer:?}: {addresses:?}",
375		);
376
377		self.kademlia_handle.add_known_peer(peer, addresses).await;
378	}

457	/// Check if the observed address is a known address.
458	fn is_known_address(known: &Multiaddr, observed: &Multiaddr) -> bool {
459		let mut known = known.iter();
460		let mut observed = observed.iter();
461
462		loop {
463			match (known.next(), observed.next()) {
464				(None, None) => return true,
465				(None, Some(Protocol::P2p(_))) => return true,
466				(Some(Protocol::P2p(_)), None) => return true,
467				(known, observed) if known != observed => return false,
468				_ => {},
469			}
470		}
471	}
472
473	/// Can `address` be added to DHT.
474	fn can_add_to_dht(address: &Multiaddr) -> bool {
475		let ip = match address.iter().next() {
476			Some(Protocol::Ip4(ip)) => IpNetwork::from(ip),
477			Some(Protocol::Ip6(ip)) => IpNetwork::from(ip),
478			Some(Protocol::Dns(_)) | Some(Protocol::Dns4(_)) | Some(Protocol::Dns6(_)) =>
479				return true,
480			_ => return false,
481		};
482
483		ip.is_global()
484	}
485
486	/// Check if `address` can be considered a new external address.
487	///
488	/// If this address replaces an older address, the expired address is returned.
489	fn is_new_external_address(
490		&mut self,
491		address: &Multiaddr,
492		peer: PeerId,
493	) -> (bool, Option<Multiaddr>) {
494		log::trace!(target: LOG_TARGET, "verify new external address: {address}");
495
496		// is the address one of our known addresses
497		if self
498			.listen_addresses
499			.read()
500			.iter()
501			.chain(self.public_addresses.iter())
502			.any(|known_address| Discovery::is_known_address(&known_address, &address))
503		{
504			return (true, None)
505		}
506
507		match self.address_confirmations.get(address) {
508			Some(confirmations) => {
509				confirmations.insert(peer);
510
511				if confirmations.len() >= MIN_ADDRESS_CONFIRMATIONS {
512					return (true, None)
513				}
514			},
515			None => {
516				let oldest = (self.address_confirmations.len() >=
517					self.address_confirmations.limiter().max_length() as usize)
518					.then(|| {
519						self.address_confirmations.pop_oldest().map(|(address, peers)| {
520							if peers.len() >= MIN_ADDRESS_CONFIRMATIONS {
521								return Some(address)
522							} else {
523								None
524							}
525						})
526					})
527					.flatten()
528					.flatten();
529
530				self.address_confirmations.insert(address.clone(), iter::once(peer).collect());
531
532				return (false, oldest)
533			},
534		}
535
536		(false, None)
537	}
538}

540impl Stream for Discovery {
541	type Item = DiscoveryEvent;
542
543	fn poll_next(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Option<Self::Item>> {
544		let this = Pin::into_inner(self);
545
546		if let Some(event) = this.pending_events.pop_front() {
547			return Poll::Ready(Some(event))
548		}
549
550		if let Some(mut delay) = this.next_kad_query.take() {
551			match delay.poll_unpin(cx) {
552				Poll::Pending => {
553					this.next_kad_query = Some(delay);
554				},
555				Poll::Ready(()) => {
556					let peer = PeerId::random();
557
558					log::trace!(target: LOG_TARGET, "start next kademlia query for {peer:?}");
559
560					match this.kademlia_handle.try_find_node(peer) {
561						Ok(query_id) => {
562							this.random_walk_query_id = Some(query_id);
563							return Poll::Ready(Some(DiscoveryEvent::RandomKademliaStarted))
564						},
565						Err(()) => {
566							this.duration_to_next_find_query = cmp::min(
567								this.duration_to_next_find_query * 2,
568								Duration::from_secs(60),
569							);
570							this.next_kad_query =
571								Some(Delay::new(this.duration_to_next_find_query));
572						},
573					}
574				},
575			}
576		}
577
578		match Pin::new(&mut this.kademlia_handle).poll_next(cx) {
579			Poll::Pending => {},
580			Poll::Ready(None) => return Poll::Ready(None),
581			Poll::Ready(Some(KademliaEvent::FindNodeSuccess { query_id, peers, .. }))
582				if Some(query_id) == this.random_walk_query_id =>
583			{
584				// the addresses are already inserted into the DHT and in `TransportManager` so
585				// there is no need to add them again. The found peers must be registered to
586				// `Peerstore` so other protocols are aware of them through `Peerset`.
587				log::trace!(target: LOG_TARGET, "dht random walk yielded {} peers", peers.len());
588
589				this.next_kad_query = Some(Delay::new(KADEMLIA_QUERY_INTERVAL));
590
591				return Poll::Ready(Some(DiscoveryEvent::RoutingTableUpdate {
592					peers: peers.into_iter().map(|(peer, _)| peer).collect(),
593				}))
594			},
595			Poll::Ready(Some(KademliaEvent::FindNodeSuccess { query_id, target, peers })) => {
596				log::trace!(target: LOG_TARGET, "find node query yielded {} peers", peers.len());
597
598				return Poll::Ready(Some(DiscoveryEvent::FindNodeSuccess {
599					query_id,
600					target,
601					peers,
602				}))
603			},
604			Poll::Ready(Some(KademliaEvent::RoutingTableUpdate { peers })) => {
605				log::trace!(target: LOG_TARGET, "routing table update, discovered {} peers", peers.len());
606
607				return Poll::Ready(Some(DiscoveryEvent::RoutingTableUpdate {
608					peers: peers.into_iter().collect(),
609				}))
610			},
611			Poll::Ready(Some(KademliaEvent::GetRecordSuccess { query_id })) => {
612				log::trace!(
613					target: LOG_TARGET,
614					"`GET_RECORD` succeeded for {query_id:?}",
615				);
616
617				return Poll::Ready(Some(DiscoveryEvent::GetRecordSuccess { query_id }));
618			},
619			Poll::Ready(Some(KademliaEvent::GetRecordPartialResult { query_id, record })) => {
620				log::trace!(
621					target: LOG_TARGET,
622					"`GET_RECORD` intermediary succeeded for {query_id:?}: {record:?}",
623				);
624
625				return Poll::Ready(Some(DiscoveryEvent::GetRecordPartialResult {
626					query_id,
627					record,
628				}));
629			},
630			Poll::Ready(Some(KademliaEvent::PutRecordSuccess { query_id, key: _ })) =>
631				return Poll::Ready(Some(DiscoveryEvent::PutRecordSuccess { query_id })),
632			Poll::Ready(Some(KademliaEvent::QueryFailed { query_id })) => {
633				match this.random_walk_query_id == Some(query_id) {
634					true => {
635						this.random_walk_query_id = None;
636						this.duration_to_next_find_query =
637							cmp::min(this.duration_to_next_find_query * 2, Duration::from_secs(60));
638						this.next_kad_query = Some(Delay::new(this.duration_to_next_find_query));
639					},
640					false => return Poll::Ready(Some(DiscoveryEvent::QueryFailed { query_id })),
641				}
642			},
643			Poll::Ready(Some(KademliaEvent::IncomingRecord { record })) => {
644				log::trace!(
645					target: LOG_TARGET,
646					"incoming `PUT_RECORD` request with key {:?} from publisher {:?}",
647					record.key,
648					record.publisher,
649				);
650
651				return Poll::Ready(Some(DiscoveryEvent::IncomingRecord { record }))
652			},
653			Poll::Ready(Some(KademliaEvent::GetProvidersSuccess {
654				provided_key,
655				providers,
656				query_id,
657			})) => {
658				log::trace!(
659					target: LOG_TARGET,
660					"`GET_PROVIDERS` for {query_id:?} with {provided_key:?} yielded {providers:?}",
661				);
662
663				return Poll::Ready(Some(DiscoveryEvent::GetProvidersSuccess {
664					query_id,
665					providers,
666				}))
667			},
668			// We do not validate incoming providers.
669			Poll::Ready(Some(KademliaEvent::IncomingProvider { .. })) => {},
670		}
671
672		match Pin::new(&mut this.identify_event_stream).poll_next(cx) {
673			Poll::Pending => {},
674			Poll::Ready(None) => return Poll::Ready(None),
675			Poll::Ready(Some(IdentifyEvent::PeerIdentified {
676				peer,
677				listen_addresses,
678				supported_protocols,
679				observed_address,
680				..
681			})) => {
682				let observed_address =
683					if let Some(Protocol::P2p(peer_id)) = observed_address.iter().last() {
684						if peer_id != *this.local_peer_id.as_ref() {
685							log::warn!(
686								target: LOG_TARGET,
687								"Discovered external address for a peer that is not us: {observed_address}",
688							);
689							None
690						} else {
691							Some(observed_address)
692						}
693					} else {
694						Some(observed_address.with(Protocol::P2p(this.local_peer_id.into())))
695					};
696
697				// Ensure that an external address with a different peer ID does not have
698				// side effects of evicting other external addresses via `ExternalAddressExpired`.
699				if let Some(observed_address) = observed_address {
700					let (is_new, expired_address) =
701						this.is_new_external_address(&observed_address, peer);
702
703					if let Some(expired_address) = expired_address {
704						log::trace!(
705							target: LOG_TARGET,
706							"Removing expired external address expired={expired_address} is_new={is_new} observed={observed_address}",
707						);
708
709						this.pending_events.push_back(DiscoveryEvent::ExternalAddressExpired {
710							address: expired_address,
711						});
712					}
713
714					if is_new {
715						this.pending_events.push_back(DiscoveryEvent::ExternalAddressDiscovered {
716							address: observed_address.clone(),
717						});
718					}
719				}
720
721				return Poll::Ready(Some(DiscoveryEvent::Identified {
722					peer,
723					listen_addresses,
724					supported_protocols,
725				}));
726			},
727		}
728
729		match Pin::new(&mut this.ping_event_stream).poll_next(cx) {
730			Poll::Pending => {},
731			Poll::Ready(None) => return Poll::Ready(None),
732			Poll::Ready(Some(PingEvent::Ping { peer, ping })) =>
733				return Poll::Ready(Some(DiscoveryEvent::Ping { peer, rtt: ping })),
734		}
735
736		if let Some(ref mut mdns_event_stream) = &mut this.mdns_event_stream {
737			match Pin::new(mdns_event_stream).poll_next(cx) {
738				Poll::Pending => {},
739				Poll::Ready(None) => return Poll::Ready(None),
740				Poll::Ready(Some(MdnsEvent::Discovered(addresses))) =>
741					return Poll::Ready(Some(DiscoveryEvent::Discovered { addresses })),
742			}
743		}
744
745		Poll::Pending
746	}
747}

749#[cfg(test)]
750mod tests {
751	use super::*;
752
753	use std::sync::atomic::AtomicU32;
754
755	use crate::{
756		config::ProtocolId,
757		peer_store::{PeerStore, PeerStoreProvider},
758	};
759	use futures::{stream::FuturesUnordered, StreamExt};
760	use sp_core::H256;
761	use sp_tracing::tracing_subscriber;
762
763	use litep2p::{
764		config::ConfigBuilder as Litep2pConfigBuilder, transport::tcp::config::Config as TcpConfig,
765		Litep2p,
766	};
767
768	#[tokio::test]
769	async fn litep2p_discovery_works() {
770		let _ = tracing_subscriber::fmt()
771			.with_env_filter(tracing_subscriber::EnvFilter::from_default_env())
772			.try_init();
773
774		let mut known_peers = HashMap::new();
775		let genesis_hash = H256::from_low_u64_be(1);
776		let fork_id = Some("test-fork-id");
777		let protocol_id = ProtocolId::from("dot");
778
779		// Build backends such that the first peer is known to all other peers.
780		let backends = (0..10)
781			.map(|i| {
782				let keypair = litep2p::crypto::ed25519::Keypair::generate();
783				let peer_id: PeerId = keypair.public().to_peer_id().into();
784
785				let listen_addresses = Arc::new(RwLock::new(HashSet::new()));
786
787				let peer_store = PeerStore::new(vec![], None);
788				let peer_store_handle: Arc<dyn PeerStoreProvider> = Arc::new(peer_store.handle());
789
790				let (discovery, ping_config, identify_config, kademlia_config, _mdns) =
791					Discovery::new(
792						peer_id,
793						&NetworkConfiguration::new_local(),
794						genesis_hash,
795						fork_id,
796						&protocol_id,
797						known_peers.clone(),
798						listen_addresses.clone(),
799						peer_store_handle,
800					);
801
802				let config = Litep2pConfigBuilder::new()
803					.with_keypair(keypair)
804					.with_tcp(TcpConfig {
805						listen_addresses: vec!["/ip6/::1/tcp/0".parse().unwrap()],
806						..Default::default()
807					})
808					.with_libp2p_ping(ping_config)
809					.with_libp2p_identify(identify_config)
810					.with_libp2p_kademlia(kademlia_config)
811					.build();
812
813				let mut litep2p = Litep2p::new(config).unwrap();
814
815				let addresses = litep2p.listen_addresses().cloned().collect::<Vec<_>>();
816				// Propagate addresses to discovery.
817				addresses.iter().for_each(|address| {
818					listen_addresses.write().insert(address.clone());
819				});
820
821				// Except the first peer, all other peers know the first peer addresses.
822				if i == 0 {
823					log::info!(target: LOG_TARGET, "First peer is {peer_id:?} with addresses {addresses:?}");
824					known_peers.insert(peer_id, addresses.clone());
825				} else {
826					let (peer, addresses) = known_peers.iter().next().unwrap();
827
828					let result = litep2p.add_known_address(*peer, addresses.into_iter().cloned());
829
830					log::info!(target: LOG_TARGET, "{peer_id:?}: Adding known peer {peer:?} with addresses {addresses:?} result={result:?}");
831
832				}
833
834				(peer_id, litep2p, discovery)
835			})
836			.collect::<Vec<_>>();
837
838		let total_peers = backends.len() as u32;
839		let remaining_peers =
840			backends.iter().map(|(peer_id, _, _)| *peer_id).collect::<HashSet<_>>();
841
842		let first_peer = *known_peers.iter().next().unwrap().0;
843
844		// Each backend must discover the whole network.
845		let mut futures = FuturesUnordered::new();
846		let num_finished = Arc::new(AtomicU32::new(0));
847
848		for (peer_id, mut litep2p, mut discovery) in backends {
849			// Remove the local peer id from the set.
850			let mut remaining_peers = remaining_peers.clone();
851			remaining_peers.remove(&peer_id);
852
853			let num_finished = num_finished.clone();
854
855			let future = async move {
856				log::info!(target: LOG_TARGET, "{peer_id:?} starting loop");
857
858				if peer_id != first_peer {
859					log::info!(target: LOG_TARGET, "{peer_id:?} dialing {first_peer:?}");
860					litep2p.dial(&first_peer).await.unwrap();
861				}
862
863				loop {
864					// We need to keep the network alive until all peers are discovered.
865					if num_finished.load(std::sync::atomic::Ordering::Relaxed) == total_peers {
866						log::info!(target: LOG_TARGET, "{peer_id:?} all peers discovered");
867						break
868					}
869
870					tokio::select! {
871						// Drive litep2p backend forward.
872						event = litep2p.next_event() => {
873							log::info!(target: LOG_TARGET, "{peer_id:?} Litep2p event: {event:?}");
874						},
875
876						// Detect discovery events.
877						event = discovery.next() => {
878							match event.unwrap() {
879								// We have discovered the peer via kademlia and established
880								// a connection on the identify protocol.
881								DiscoveryEvent::Identified { peer, .. } => {
882									log::info!(target: LOG_TARGET, "{peer_id:?} Peer {peer} identified");
883
884									remaining_peers.remove(&peer);
885
886									if remaining_peers.is_empty() {
887										log::info!(target: LOG_TARGET, "{peer_id:?} All peers discovered");
888
889										num_finished.fetch_add(1, std::sync::atomic::Ordering::AcqRel);
890									}
891								},
892
893								event => {
894									log::info!(target: LOG_TARGET, "{peer_id:?} Discovery event: {event:?}");
895								}
896							}
897						}
898					}
899				}
900			};
901
902			futures.push(future);
903		}
904
905		// Futures will exit when all peers are discovered.
906		tokio::time::timeout(Duration::from_secs(60), futures.next())
907			.await
908			.expect("All peers should finish within 60 seconds");
909	}
