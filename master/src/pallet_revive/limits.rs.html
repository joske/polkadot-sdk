18//! Limits that are observeable by contract code.
19//!
20//! It is important to never change this limits without supporting the old limits
21//! for already deployed contracts. This is what the [`crate::Contract::behaviour_version`]
22//! is meant for. This is true for either increasing or decreasing the limit.
23//!
24//! Limits in this file are different from the limits configured on the [`Config`] trait which are
25//! generally only affect actions that cannot be performed by a contract: For example things related
26//! to deposits and weights are allowed to be changed as they are paid by root callers which
27//! are not contracts.
28//!
29//! Exceptions to this rule apply: Limits in the [`code`] module can be increased
30//! without emulating the old values for existing contracts. Reason is that those limits are only
31//! applied **once** at code upload time. Since this action cannot be performed by contracts we
32//! can change those limits without breaking existing contracts. Please keep in mind that we should
33//! only ever **increase** those values but never decrease.
34
35/// The maximum depth of the call stack.
36///
37/// A 0 means that no callings of other contracts are possible. In other words only the origin
38/// called "root contract" is allowed to execute then.
39pub const CALL_STACK_DEPTH: u32 = 5;
40
41/// The maximum number of topics a call to [`crate::SyscallDoc::deposit_event`] can emit.
42///
43/// We set it to the same limit that ethereum has. It is unlikely to change.
44pub const NUM_EVENT_TOPICS: u32 = 4;
45
46/// Maximum size of events (including topics) and storage values.
47pub const PAYLOAD_BYTES: u32 = 416;
48
49/// The maximum size of the transient storage in bytes.
50///
51/// This includes keys, values, and previous entries used for storage rollback.
52pub const TRANSIENT_STORAGE_BYTES: u32 = 4 * 1024;
53
54/// The maximum allowable length in bytes for (transient) storage keys.
55pub const STORAGE_KEY_BYTES: u32 = 128;
56
57/// The page size in which PolkaVM should allocate memory chunks.
58pub const PAGE_SIZE: u32 = 4 * 1024;
59
60/// The maximum amount of immutable bytes a single contract can store.
61///
62/// The current limit of 4kb allows storing up 16 U256 immutable variables.
63/// Which should always be enough because Solidity allows for 16 local (stack) variables.
64pub const IMMUTABLE_BYTES: u32 = 4 * 1024;
65
66/// Limits that are only enforced on code upload.
67///
68/// # Note
69///
70/// This limit can be increased later without breaking existing contracts
71/// as it is only enforced at code upload time. Code already uploaded
72/// will not be affected by those limits.
73pub mod code {
74	use super::PAGE_SIZE;
75	use crate::{CodeVec, Config, Error, LOG_TARGET};
76	use alloc::vec::Vec;
77	use sp_runtime::DispatchError;
78
79	/// The maximum length of a code blob in bytes.
80	///
81	/// This mostly exist to prevent parsing too big blobs and to
82	/// have a maximum encoded length. The actual memory calculation
83	/// is purely based off [`STATIC_MEMORY_BYTES`].
84	pub const BLOB_BYTES: u32 = 256 * 1024;
85
86	/// Maximum size the program is allowed to take in memory.
87	///
88	/// This includes data and code. Increasing this limit will allow
89	/// for more code or more data. However, since code will decompress
90	/// into a bigger representation on compilation it will only increase
91	/// the allowed code size by [`BYTE_PER_INSTRUCTION`].
92	pub const STATIC_MEMORY_BYTES: u32 = 2 * 1024 * 1024;
93
94	/// The maximum size of a basic block in number of instructions.
95	///
96	/// We need to limit the size of basic blocks because the interpreters lazy compilation
97	/// compiles one basic block at a time. A malicious program could trigger the compilation
98	/// of the whole program by creating one giant basic block otherwise.
99	pub const BASIC_BLOCK_SIZE: u32 = 1000;
100
101	/// How much memory each instruction will take in-memory after compilation.
102	///
103	/// This is `size_of<usize>() + 16`. But we don't use `usize` here so it isn't
104	/// different on the native runtime (used for testing).
105	pub const BYTES_PER_INSTRUCTION: u32 = 20;
106
107	/// The code is stored multiple times as part of the compiled program.
108	const EXTRA_OVERHEAD_PER_CODE_BYTE: u32 = 4;
109
110	/// Make sure that the various program parts are within the defined limits.
111	pub fn enforce<T: Config>(
112		blob: Vec<u8>,
113		available_syscalls: &[&[u8]],
114	) -> Result<CodeVec, DispatchError> {
115		fn round_page(n: u32) -> u64 {
116			// performing the rounding in u64 in order to prevent overflow
117			u64::from(n).next_multiple_of(PAGE_SIZE.into())
118		}
119
120		let blob: CodeVec = blob.try_into().map_err(|_| <Error<T>>::BlobTooLarge)?;
121
122		#[cfg(feature = "std")]
123		if std::env::var_os("REVIVE_SKIP_VALIDATION").is_some() {
124			log::warn!(target: LOG_TARGET, "Skipping validation because env var REVIVE_SKIP_VALIDATION is set");
125			return Ok(blob)
126		}
127
128		let program = polkavm::ProgramBlob::parse(blob.as_slice().into()).map_err(|err| {
129			log::debug!(target: LOG_TARGET, "failed to parse polkavm blob: {err:?}");
130			Error::<T>::CodeRejected
131		})?;
132
133		if !program.is_64_bit() {
134			log::debug!(target: LOG_TARGET, "32bit programs are not supported.");
135			Err(Error::<T>::CodeRejected)?;
136		}
137
138		// Need to check that no non-existent syscalls are used. This allows us to add
139		// new syscalls later without affecting already deployed code.
140		for (idx, import) in program.imports().iter().enumerate() {
141			// We are being defensive in case an attacker is able to somehow include
142			// a lot of imports. This is important because we search the array of host
143			// functions for every import.
144			if idx == available_syscalls.len() {
145				log::debug!(target: LOG_TARGET, "Program contains too many imports.");
146				Err(Error::<T>::CodeRejected)?;
147			}
148			let Some(import) = import else {
149				log::debug!(target: LOG_TARGET, "Program contains malformed import.");
150				return Err(Error::<T>::CodeRejected.into());
151			};
152			if !available_syscalls.contains(&import.as_bytes()) {
153				log::debug!(target: LOG_TARGET, "Program references unknown syscall: {}", import);
154				Err(Error::<T>::CodeRejected)?;
155			}
156		}
157
158		// This scans the whole program but we only do it once on code deployment.
159		// It is safe to do unchecked math in u32 because the size of the program
160		// was already checked above.
161		use polkavm::program::ISA64_V1 as ISA;
162		let mut num_instructions: u32 = 0;
163		let mut max_basic_block_size: u32 = 0;
164		let mut basic_block_size: u32 = 0;
165		for inst in program.instructions(ISA) {
166			use polkavm::program::Instruction;
167			num_instructions += 1;
168			basic_block_size += 1;
169			if inst.kind.opcode().starts_new_basic_block() {
170				max_basic_block_size = max_basic_block_size.max(basic_block_size);
171				basic_block_size = 0;
172			}
173			match inst.kind {
174				Instruction::invalid => {
175					log::debug!(target: LOG_TARGET, "invalid instruction at offset {}", inst.offset);
176					return Err(<Error<T>>::InvalidInstruction.into())
177				},
178				Instruction::sbrk(_, _) => {
179					log::debug!(target: LOG_TARGET, "sbrk instruction is not allowed. offset {}", inst.offset);
180					return Err(<Error<T>>::InvalidInstruction.into())
181				},
182				// Only benchmarking code is allowed to circumvent the import table. We might want
183				// to remove this magic syscall number later. Hence we need to prevent contracts
184				// from using it.
185				#[cfg(not(feature = "runtime-benchmarks"))]
186				Instruction::ecalli(idx) if idx == crate::SENTINEL => {
187					log::debug!(target: LOG_TARGET, "reserved syscall idx {idx}. offset {}", inst.offset);
188					return Err(<Error<T>>::InvalidInstruction.into())
189				},
190				_ => (),
191			}
192		}
193
194		if max_basic_block_size > BASIC_BLOCK_SIZE {
195			log::debug!(target: LOG_TARGET, "basic block too large: {max_basic_block_size} limit: {BASIC_BLOCK_SIZE}");
196			return Err(Error::<T>::BasicBlockTooLarge.into())
197		}
198
199		// The memory consumptions is the byte size of the whole blob,
200		// minus the RO data payload in the blob,
201		// minus the RW data payload in the blob,
202		// plus the RO data in memory (which is always equal or bigger than the RO payload),
203		// plus RW data in memory, plus stack size in memory.
204		// plus the overhead of instructions in memory which is derived from the code
205		// size itself and the number of instruction
206		let memory_size = (blob.len() as u64)
207			.saturating_add(round_page(program.ro_data_size()))
208			.saturating_sub(program.ro_data().len() as u64)
209			.saturating_add(round_page(program.rw_data_size()))
210			.saturating_sub(program.rw_data().len() as u64)
211			.saturating_add(round_page(program.stack_size()))
212			.saturating_add(
213				u64::from(num_instructions).saturating_mul(BYTES_PER_INSTRUCTION.into()),
214			)
215			.saturating_add(
216				(program.code().len() as u64).saturating_mul(EXTRA_OVERHEAD_PER_CODE_BYTE.into()),
217			);
218
219		if memory_size > STATIC_MEMORY_BYTES.into() {
220			log::debug!(target: LOG_TARGET, "static memory too large: {memory_size} limit: {STATIC_MEMORY_BYTES}");
221			return Err(Error::<T>::StaticMemoryTooLarge.into())
222		}
223
224		Ok(blob)
225	}
