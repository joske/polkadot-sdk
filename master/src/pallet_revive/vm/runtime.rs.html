41/// Abstraction over the memory access within syscalls.
42///
43/// The reason for this abstraction is that we run syscalls on the host machine when
44/// benchmarking them. In that case we have direct access to the contract's memory. However, when
45/// running within PolkaVM we need to resort to copying as we can't map the contracts memory into
46/// the host (as of now).
47pub trait Memory<T: Config> {
48	/// Read designated chunk from the sandbox memory into the supplied buffer.
49	///
50	/// Returns `Err` if one of the following conditions occurs:
51	///
52	/// - requested buffer is not within the bounds of the sandbox memory.
53	fn read_into_buf(&self, ptr: u32, buf: &mut [u8]) -> Result<(), DispatchError>;
54
55	/// Write the given buffer to the designated location in the sandbox memory.
56	///
57	/// Returns `Err` if one of the following conditions occurs:
58	///
59	/// - designated area is not within the bounds of the sandbox memory.
60	fn write(&mut self, ptr: u32, buf: &[u8]) -> Result<(), DispatchError>;
61
62	/// Zero the designated location in the sandbox memory.
63	///
64	/// Returns `Err` if one of the following conditions occurs:
65	///
66	/// - designated area is not within the bounds of the sandbox memory.
67	fn zero(&mut self, ptr: u32, len: u32) -> Result<(), DispatchError>;
68
69	/// Read designated chunk from the sandbox memory.
70	///
71	/// Returns `Err` if one of the following conditions occurs:
72	///
73	/// - requested buffer is not within the bounds of the sandbox memory.
74	fn read(&self, ptr: u32, len: u32) -> Result<Vec<u8>, DispatchError> {
75		let mut buf = vec![0u8; len as usize];
76		self.read_into_buf(ptr, buf.as_mut_slice())?;
77		Ok(buf)
78	}
79
80	/// Same as `read` but reads into a fixed size buffer.
81	fn read_array<const N: usize>(&self, ptr: u32) -> Result<[u8; N], DispatchError> {
82		let mut buf = [0u8; N];
83		self.read_into_buf(ptr, &mut buf)?;
84		Ok(buf)
85	}
86
87	/// Read a `u32` from the sandbox memory.
88	fn read_u32(&self, ptr: u32) -> Result<u32, DispatchError> {
89		let buf: [u8; 4] = self.read_array(ptr)?;
90		Ok(u32::from_le_bytes(buf))
91	}
92
93	/// Read a `U256` from the sandbox memory.
94	fn read_u256(&self, ptr: u32) -> Result<U256, DispatchError> {
95		let buf: [u8; 32] = self.read_array(ptr)?;
96		Ok(U256::from_little_endian(&buf))
97	}
98
99	/// Read a `H160` from the sandbox memory.
100	fn read_h160(&self, ptr: u32) -> Result<H160, DispatchError> {
101		let mut buf = H160::default();
102		self.read_into_buf(ptr, buf.as_bytes_mut())?;
103		Ok(buf)
104	}
105
106	/// Read a `H256` from the sandbox memory.
107	fn read_h256(&self, ptr: u32) -> Result<H256, DispatchError> {
108		let mut code_hash = H256::default();
109		self.read_into_buf(ptr, code_hash.as_bytes_mut())?;
110		Ok(code_hash)
111	}
112}

126// Memory implementation used in benchmarking where guest memory is mapped into the host.
127//
128// Please note that we could optimize the `read_as_*` functions by decoding directly from
129// memory without a copy. However, we don't do that because as it would change the behaviour
130// of those functions: A `read_as` with a `len` larger than the actual type can succeed
131// in the streaming implementation while it could fail with a segfault in the copy implementation.
132#[cfg(feature = "runtime-benchmarks")]
133impl<T: Config> Memory<T> for [u8] {
134	fn read_into_buf(&self, ptr: u32, buf: &mut [u8]) -> Result<(), DispatchError> {
135		let ptr = ptr as usize;
136		let bound_checked =
137			self.get(ptr..ptr + buf.len()).ok_or_else(|| Error::<T>::OutOfBounds)?;
138		buf.copy_from_slice(bound_checked);
139		Ok(())
140	}
141
142	fn write(&mut self, ptr: u32, buf: &[u8]) -> Result<(), DispatchError> {
143		let ptr = ptr as usize;
144		let bound_checked =
145			self.get_mut(ptr..ptr + buf.len()).ok_or_else(|| Error::<T>::OutOfBounds)?;
146		bound_checked.copy_from_slice(buf);
147		Ok(())
148	}
149
150	fn zero(&mut self, ptr: u32, len: u32) -> Result<(), DispatchError> {
151		<[u8] as Memory<T>>::write(self, ptr, &vec![0; len as usize])
152	}
153}

155impl<T: Config> Memory<T> for polkavm::RawInstance {
156	fn read_into_buf(&self, ptr: u32, buf: &mut [u8]) -> Result<(), DispatchError> {
157		self.read_memory_into(ptr, buf)
158			.map(|_| ())
159			.map_err(|_| Error::<T>::OutOfBounds.into())
160	}
161
162	fn write(&mut self, ptr: u32, buf: &[u8]) -> Result<(), DispatchError> {
163		self.write_memory(ptr, buf).map_err(|_| Error::<T>::OutOfBounds.into())
164	}
165
166	fn zero(&mut self, ptr: u32, len: u32) -> Result<(), DispatchError> {
167		self.zero_memory(ptr, len).map_err(|_| Error::<T>::OutOfBounds.into())
168	}
169}

216/// Enumerates all possible reasons why a trap was generated.
217///
218/// This is either used to supply the caller with more information about why an error
219/// occurred (the SupervisorError variant).
220/// The other case is where the trap does not constitute an error but rather was invoked
221/// as a quick way to terminate the application (all other variants).
222#[derive(RuntimeDebug)]
223pub enum TrapReason {
224	/// The supervisor trapped the contract because of an error condition occurred during
225	/// execution in privileged code.
226	SupervisorError(DispatchError),
227	/// Signals that trap was generated in response to call `seal_return` host function.
228	Return(ReturnData),
229	/// Signals that a trap was generated in response to a successful call to the
230	/// `seal_terminate` host function.
231	Termination,
232}
233
234impl<T: Into<DispatchError>> From<T> for TrapReason {
235	fn from(from: T) -> Self {
236		Self::SupervisorError(from.into())
237	}
238}

387/// For functions that modify storage, benchmarks are performed with one item in the
388/// storage. To account for the worst-case scenario, the weight of the overhead of
389/// writing to or reading from full storage is included. For transient storage writes,
390/// the rollback weight is added to reflect the worst-case scenario for this operation.
391macro_rules! cost_storage {
392    (write_transient, $name:ident $(, $arg:expr )*) => {
393        T::WeightInfo::$name($( $arg ),*)
394            .saturating_add(T::WeightInfo::rollback_transient_storage())
395            .saturating_add(T::WeightInfo::set_transient_storage_full()
396            .saturating_sub(T::WeightInfo::set_transient_storage_empty()))
397    };
398
399    (read_transient, $name:ident $(, $arg:expr )*) => {
400        T::WeightInfo::$name($( $arg ),*)
401            .saturating_add(T::WeightInfo::get_transient_storage_full()
402            .saturating_sub(T::WeightInfo::get_transient_storage_empty()))
403    };
404
405    (write, $name:ident $(, $arg:expr )*) => {
406        T::WeightInfo::$name($( $arg ),*)
407            .saturating_add(T::WeightInfo::set_storage_full()
408            .saturating_sub(T::WeightInfo::set_storage_empty()))
409    };
410
411    (read, $name:ident $(, $arg:expr )*) => {
412        T::WeightInfo::$name($( $arg ),*)
413            .saturating_add(T::WeightInfo::get_storage_full()
414            .saturating_sub(T::WeightInfo::get_storage_empty()))
415    };
416}
417
418macro_rules! cost_args {
419	// cost_args!(name, a, b, c) -> T::WeightInfo::name(a, b, c).saturating_sub(T::WeightInfo::name(0, 0, 0))
420	($name:ident, $( $arg: expr ),+) => {
421		(T::WeightInfo::$name($( $arg ),+).saturating_sub(cost_args!(@call_zero $name, $( $arg ),+)))
422	};
423	// Transform T::WeightInfo::name(a, b, c) into T::WeightInfo::name(0, 0, 0)
424	(@call_zero $name:ident, $( $arg:expr ),*) => {
425		T::WeightInfo::$name($( cost_args!(@replace_token $arg) ),*)
426	};
427	// Replace the token with 0.
428	(@replace_token $_in:tt) => { 0 };
429}

514			Bn128Mul => T::WeightInfo::bn128_mul(),
515			Bn128Pairing(len) => T::WeightInfo::bn128_pairing(len),
516			Identity(len) => T::WeightInfo::identity(len),
517			Blake2F(rounds) => T::WeightInfo::blake2f(rounds),
518			Modexp(gas) => {
519				use frame_support::weights::constants::WEIGHT_REF_TIME_PER_SECOND;
520				/// Current approximation of the gas/s consumption considering
521				/// EVM execution over compiled WASM (on 4.4Ghz CPU).
522				/// Given the 2000ms Weight, from which 75% only are used for transactions,
523				/// the total EVM execution gas limit is: GAS_PER_SECOND * 2 * 0.75 ~= 60_000_000.
524				const GAS_PER_SECOND: u64 = 40_000_000;
525
526				/// Approximate ratio of the amount of Weight per Gas.
527				/// u64 works for approximations because Weight is a very small unit compared to
528				/// gas.
529				const WEIGHT_PER_GAS: u64 = WEIGHT_REF_TIME_PER_SECOND / GAS_PER_SECOND;
530				Weight::from_parts(gas.saturating_mul(WEIGHT_PER_GAS), 0)
531			},
532		}
533	}

607	pub fn handle_interrupt(
608		&mut self,
609		interrupt: Result<polkavm::InterruptKind, polkavm::Error>,
610		module: &polkavm::Module,
611		instance: &mut M,
612	) -> Option<ExecResult> {
613		use polkavm::InterruptKind::*;
614
615		match interrupt {
616			Err(error) => {
617				// in contrast to the other returns this "should" not happen: log level error
618				log::error!(target: LOG_TARGET, "polkavm execution error: {error}");
619				Some(Err(Error::<E::T>::ExecutionFailed.into()))
620			},
621			Ok(Finished) =>
622				Some(Ok(ExecReturnValue { flags: ReturnFlags::empty(), data: Vec::new() })),
623			Ok(Trap) => Some(Err(Error::<E::T>::ContractTrapped.into())),
624			Ok(Segfault(_)) => Some(Err(Error::<E::T>::ExecutionFailed.into())),
625			Ok(NotEnoughGas) => Some(Err(Error::<E::T>::OutOfGas.into())),
626			Ok(Step) => None,
627			Ok(Ecalli(idx)) => {
628				// This is a special hard coded syscall index which is used by benchmarks
629				// to abort contract execution. It is used to terminate the execution without
630				// breaking up a basic block. The fixed index is used so that the benchmarks
631				// don't have to deal with import tables.
632				if cfg!(feature = "runtime-benchmarks") && idx == SENTINEL {
633					return Some(Ok(ExecReturnValue {
634						flags: ReturnFlags::empty(),
635						data: Vec::new(),
636					}))
637				}
638				let Some(syscall_symbol) = module.imports().get(idx) else {
639					return Some(Err(<Error<E::T>>::InvalidSyscall.into()));
640				};
641				match self.handle_ecall(instance, syscall_symbol.as_bytes()) {
642					Ok(None) => None,
643					Ok(Some(return_value)) => {
644						instance.write_output(return_value);
645						None
646					},
647					Err(TrapReason::Return(ReturnData { flags, data })) =>
648						match ReturnFlags::from_bits(flags) {
649							None => Some(Err(Error::<E::T>::InvalidCallFlags.into())),
650							Some(flags) => Some(Ok(ExecReturnValue { flags, data })),
651						},
652					Err(TrapReason::Termination) => Some(Ok(Default::default())),
653					Err(TrapReason::SupervisorError(error)) => Some(Err(error.into())),
654				}
655			},
656		}
657	}

685	/// Write the given buffer and its length to the designated locations in sandbox memory and
686	/// charge gas according to the token returned by `create_token`.
687	///
688	/// `out_ptr` is the location in sandbox memory where `buf` should be written to.
689	/// `out_len_ptr` is an in-out location in sandbox memory. It is read to determine the
690	/// length of the buffer located at `out_ptr`. If that buffer is smaller than the actual
691	/// `buf.len()`, only what fits into that buffer is written to `out_ptr`.
692	/// The actual amount of bytes copied to `out_ptr` is written to `out_len_ptr`.
693	///
694	/// If `out_ptr` is set to the sentinel value of `SENTINEL` and `allow_skip` is true the
695	/// operation is skipped and `Ok` is returned. This is supposed to help callers to make copying
696	/// output optional. For example to skip copying back the output buffer of an `seal_call`
697	/// when the caller is not interested in the result.
698	///
699	/// `create_token` can optionally instruct this function to charge the gas meter with the token
700	/// it returns. `create_token` receives the variable amount of bytes that are about to be copied
701	/// by this function.
702	///
703	/// In addition to the error conditions of `Memory::write` this functions returns
704	/// `Err` if the size of the buffer located at `out_ptr` is too small to fit `buf`.
705	pub fn write_sandbox_output(
706		&mut self,
707		memory: &mut M,
708		out_ptr: u32,
709		out_len_ptr: u32,
710		buf: &[u8],
711		allow_skip: bool,
712		create_token: impl FnOnce(u32) -> Option<RuntimeCosts>,
713	) -> Result<(), DispatchError> {
714		if allow_skip && out_ptr == SENTINEL {
715			return Ok(());
716		}
717
718		let len = memory.read_u32(out_len_ptr)?;
719		let buf_len = len.min(buf.len() as u32);
720
721		if let Some(costs) = create_token(buf_len) {
722			self.charge_gas(costs)?;
723		}
724
725		memory.write(out_ptr, &buf[..buf_len as usize])?;
726		memory.write(out_len_ptr, &buf_len.encode())
727	}

729	/// Same as `write_sandbox_output` but for static size output.

731		&mut self,
732		memory: &mut M,

734		buf: &[u8],
735		allow_skip: bool,
736		create_token: impl FnOnce(u32) -> Option<RuntimeCosts>,
737	) -> Result<(), DispatchError> {
738		if buf.is_empty() || (allow_skip && out_ptr == SENTINEL) {
739			return Ok(());
740		}

742		let buf_len = buf.len() as u32;
743		if let Some(costs) = create_token(buf_len) {
744			self.charge_gas(costs)?;
745		}

747		memory.write(out_ptr, buf)
748	}

750	/// Computes the given hash function on the supplied input.
751	///
752	/// Reads from the sandboxed input buffer into an intermediate buffer.
753	/// Returns the result directly to the output buffer of the sandboxed memory.
754	///
755	/// It is the callers responsibility to provide an output buffer that
756	/// is large enough to hold the expected amount of bytes returned by the
757	/// chosen hash function.
758	///
759	/// # Note
760	///
761	/// The `input` and `output` buffers may overlap.
762	fn compute_hash_on_intermediate_buffer<F, R>(
763		&self,
764		memory: &mut M,
765		hash_fn: F,
766		input_ptr: u32,
767		input_len: u32,
768		output_ptr: u32,
769	) -> Result<(), DispatchError>
770	where
771		F: FnOnce(&[u8]) -> R,
772		R: AsRef<[u8]>,
773	{
774		// Copy input into supervisor memory.
775		let input = memory.read(input_ptr, input_len)?;
776		// Compute the hash on the input buffer using the given hash function.
777		let hash = hash_fn(&input);
778		// Write the resulting hash back into the sandboxed output buffer.
779		memory.write(output_ptr, hash.as_ref())?;
780		Ok(())
781	}

783	/// Fallible conversion of a `ExecError` to `ReturnErrorCode`.
784	///
785	/// This is used when converting the error returned from a subcall in order to decide
786	/// whether to trap the caller or allow handling of the error.
787	fn exec_error_into_return_code(from: ExecError) -> Result<ReturnErrorCode, DispatchError> {
788		use crate::exec::ErrorOrigin::Callee;
789		use ReturnErrorCode::*;
790
791		let transfer_failed = Error::<E::T>::TransferFailed.into();
792		let out_of_gas = Error::<E::T>::OutOfGas.into();
793		let out_of_deposit = Error::<E::T>::StorageDepositLimitExhausted.into();
794		let duplicate_contract = Error::<E::T>::DuplicateContract.into();
795		let unsupported_precompile = Error::<E::T>::UnsupportedPrecompileAddress.into();
796
797		// errors in the callee do not trap the caller
798		match (from.error, from.origin) {
799			(err, _) if err == transfer_failed => Ok(TransferFailed),
800			(err, _) if err == duplicate_contract => Ok(DuplicateContractAddress),
801			(err, _) if err == unsupported_precompile => Err(err),
802			(err, Callee) if err == out_of_gas || err == out_of_deposit => Ok(OutOfResources),
803			(_, Callee) => Ok(CalleeTrapped),
804			(err, _) => Err(err),
805		}
806	}

831	fn set_storage(
832		&mut self,
833		memory: &M,
834		flags: u32,
835		key_ptr: u32,
836		key_len: u32,
837		value: StorageValue,
838	) -> Result<u32, TrapReason> {
839		let transient = Self::is_transient(flags)?;
840		let costs = |new_bytes: u32, old_bytes: u32| {
841			if transient {
842				RuntimeCosts::SetTransientStorage { new_bytes, old_bytes }
843			} else {
844				RuntimeCosts::SetStorage { new_bytes, old_bytes }
845			}
846		};
847
848		let value_len = match &value {
849			StorageValue::Memory { ptr: _, len } => *len,
850			StorageValue::Value(data) => data.len() as u32,
851		};
852
853		let max_size = self.ext.max_value_size();
854		let charged = self.charge_gas(costs(value_len, self.ext.max_value_size()))?;
855		if value_len > max_size {
856			return Err(Error::<E::T>::ValueTooLarge.into());
857		}
858
859		let key = self.decode_key(memory, key_ptr, key_len)?;
860
861		let value = match value {
862			StorageValue::Memory { ptr, len } => Some(memory.read(ptr, len)?),
863			StorageValue::Value(data) => Some(data),
864		};
865
866		let write_outcome = if transient {
867			self.ext.set_transient_storage(&key, value, false)?
868		} else {
869			self.ext.set_storage(&key, value, false)?
870		};
871
872		self.adjust_gas(charged, costs(value_len, write_outcome.old_len()));
873		Ok(write_outcome.old_len_with_sentinel())
874	}

902	fn get_storage(
903		&mut self,
904		memory: &mut M,
905		flags: u32,
906		key_ptr: u32,
907		key_len: u32,
908		out_ptr: u32,
909		read_mode: StorageReadMode,
910	) -> Result<ReturnErrorCode, TrapReason> {
911		let transient = Self::is_transient(flags)?;
912		let costs = |len| {
913			if transient {
914				RuntimeCosts::GetTransientStorage(len)
915			} else {
916				RuntimeCosts::GetStorage(len)
917			}
918		};
919		let charged = self.charge_gas(costs(self.ext.max_value_size()))?;
920		let key = self.decode_key(memory, key_ptr, key_len)?;
921		let outcome = if transient {
922			self.ext.get_transient_storage(&key)
923		} else {
924			self.ext.get_storage(&key)
925		};
926
927		if let Some(value) = outcome {
928			self.adjust_gas(charged, costs(value.len() as u32));
929
930			match read_mode {
931				StorageReadMode::FixedOutput32 => {
932					let mut fixed_output = [0u8; 32];
933					let len = value.len().min(fixed_output.len());

936					self.write_fixed_sandbox_output(
937						memory,
938						out_ptr,
939						&fixed_output,
940						false,
941						already_charged,
942					)?;
943					Ok(ReturnErrorCode::Success)
944				},
945				StorageReadMode::VariableOutput { output_len_ptr: out_len_ptr } => {
946					self.write_sandbox_output(
947						memory,
948						out_ptr,
949						out_len_ptr,
950						&value,
951						false,
952						already_charged,
953					)?;
954					Ok(ReturnErrorCode::Success)
955				},
956			}
957		} else {
958			self.adjust_gas(charged, costs(0));
959
960			match read_mode {
961				StorageReadMode::FixedOutput32 => {
962					self.write_fixed_sandbox_output(
963						memory,
964						out_ptr,
965						&[0u8; 32],
966						false,
967						already_charged,
968					)?;
969					Ok(ReturnErrorCode::Success)
970				},
971				StorageReadMode::VariableOutput { .. } => Ok(ReturnErrorCode::KeyNotFound),
972			}
973		}

1044	fn call(
1045		&mut self,
1046		memory: &mut M,
1047		flags: CallFlags,
1048		call_type: CallType,
1049		callee_ptr: u32,
1050		deposit_ptr: u32,
1051		weight: Weight,
1052		input_data_ptr: u32,
1053		input_data_len: u32,
1054		output_ptr: u32,
1055		output_len_ptr: u32,
1056	) -> Result<ReturnErrorCode, TrapReason> {
1057		let callee = memory.read_h160(callee_ptr)?;
1058		let precompile = <AllPrecompiles<E::T>>::get::<E>(&callee.as_fixed_bytes());
1059		match &precompile {
1060			Some(precompile) if precompile.has_contract_info() =>
1061				self.charge_gas(RuntimeCosts::PrecompileWithInfoBase)?,
1062			Some(_) => self.charge_gas(RuntimeCosts::PrecompileBase)?,
1063			None => self.charge_gas(call_type.cost())?,
1064		};
1065
1066		let deposit_limit = memory.read_u256(deposit_ptr)?;
1067
1068		let input_data = if flags.contains(CallFlags::CLONE_INPUT) {
1069			let input = self.input_data.as_ref().ok_or(Error::<E::T>::InputForwarded)?;
1070			charge_gas!(self, RuntimeCosts::CallInputCloned(input.len() as u32))?;
1071			input.clone()
1072		} else if flags.contains(CallFlags::FORWARD_INPUT) {
1073			self.input_data.take().ok_or(Error::<E::T>::InputForwarded)?
1074		} else {
1075			if precompile.is_some() {
1076				self.charge_gas(RuntimeCosts::PrecompileDecode(input_data_len))?;
1077			} else {
1078				self.charge_gas(RuntimeCosts::CopyFromContract(input_data_len))?;
1079			}
1080			memory.read(input_data_ptr, input_data_len)?
1081		};
1082
1083		let call_outcome = match call_type {
1084			CallType::Call { value_ptr } => {
1085				let read_only = flags.contains(CallFlags::READ_ONLY);
1086				let value = memory.read_u256(value_ptr)?;
1087				if value > 0u32.into() {
1088					// If the call value is non-zero and state change is not allowed, issue an
1089					// error.
1090					if read_only || self.ext.is_read_only() {
1091						return Err(Error::<E::T>::StateChangeDenied.into());
1092					}
1093					self.charge_gas(RuntimeCosts::CallTransferSurcharge)?;
1094				}
1095				self.ext.call(
1096					weight,
1097					deposit_limit,
1098					&callee,
1099					value,
1100					input_data,
1101					flags.contains(CallFlags::ALLOW_REENTRY),
1102					read_only,
1103				)
1104			},
1105			CallType::DelegateCall => {
1106				if flags.intersects(CallFlags::ALLOW_REENTRY | CallFlags::READ_ONLY) {
1107					return Err(Error::<E::T>::InvalidCallFlags.into());
1108				}
1109				self.ext.delegate_call(weight, deposit_limit, callee, input_data)
1110			},
1111		};
1112
1113		match call_outcome {
1114			// `TAIL_CALL` only matters on an `OK` result. Otherwise the call stack comes to
1115			// a halt anyways without anymore code being executed.
1116			Ok(_) if flags.contains(CallFlags::TAIL_CALL) => {
1117				let output = mem::take(self.ext.last_frame_output_mut());
1118				return Err(TrapReason::Return(ReturnData {
1119					flags: output.flags.bits(),
1120					data: output.data,
1121				}));
1122			},
1123			Ok(_) => {
1124				let output = mem::take(self.ext.last_frame_output_mut());
1125				let write_result = self.write_sandbox_output(
1126					memory,
1127					output_ptr,
1128					output_len_ptr,
1129					&output.data,
1130					true,
1131					|len| Some(RuntimeCosts::CopyToContract(len)),
1132				);
1133				*self.ext.last_frame_output_mut() = output;
1134				write_result?;
1135				Ok(self.ext.last_frame_output().into())
1136			},
1137			Err(err) => {
1138				let error_code = Self::exec_error_into_return_code(err)?;
1139				memory.write(output_len_ptr, &0u32.to_le_bytes())?;
1140				Ok(error_code)
1141			},
1142		}
1143	}

1145	fn instantiate(
1146		&mut self,
1147		memory: &mut M,
1148		code_hash_ptr: u32,
1149		weight: Weight,
1150		deposit_ptr: u32,
1151		value_ptr: u32,
1152		input_data_ptr: u32,
1153		input_data_len: u32,
1154		address_ptr: u32,
1155		output_ptr: u32,
1156		output_len_ptr: u32,
1157		salt_ptr: u32,
1158	) -> Result<ReturnErrorCode, TrapReason> {
1159		self.charge_gas(RuntimeCosts::Instantiate { input_data_len })?;
1160		let deposit_limit: U256 = memory.read_u256(deposit_ptr)?;
1161		let value = memory.read_u256(value_ptr)?;
1162		let code_hash = memory.read_h256(code_hash_ptr)?;
1163		let input_data = memory.read(input_data_ptr, input_data_len)?;
1164		let salt = if salt_ptr == SENTINEL {
1165			None
1166		} else {
1167			let salt: [u8; 32] = memory.read_array(salt_ptr)?;
1168			Some(salt)
1169		};
1170
1171		match self.ext.instantiate(
1172			weight,
1173			deposit_limit,
1174			code_hash,
1175			value,
1176			input_data,
1177			salt.as_ref(),
1178		) {
1179			Ok(address) => {
1180				if !self.ext.last_frame_output().flags.contains(ReturnFlags::REVERT) {
1181					self.write_fixed_sandbox_output(
1182						memory,
1183						address_ptr,
1184						&address.as_bytes(),
1185						true,
1186						already_charged,
1187					)?;
1188				}
1189				let output = mem::take(self.ext.last_frame_output_mut());
1190				let write_result = self.write_sandbox_output(
1191					memory,
1192					output_ptr,
1193					output_len_ptr,
1194					&output.data,
1195					true,
1196					|len| Some(RuntimeCosts::CopyToContract(len)),
1197				);
1198				*self.ext.last_frame_output_mut() = output;
1199				write_result?;
1200				Ok(self.ext.last_frame_output().into())
1201			},
1202			Err(err) => Ok(Self::exec_error_into_return_code(err)?),
1203		}
1204	}

1428	/// See [`pallet_revive_uapi::HostFn::call_data_copy`].

1431		&mut self,
1432		memory: &mut M,
1433		out_ptr: u32,
1434		out_len: u32,
1435		offset: u32,
1436	) -> Result<(), TrapReason> {
1437		self.charge_gas(RuntimeCosts::CallDataCopy(out_len))?;
1438
1439		let Some(input) = self.input_data.as_ref() else {
1440			return Err(Error::<E::T>::InputForwarded.into());
1441		};
1442
1443		let start = offset as usize;
1444		if start >= input.len() {
1445			memory.zero(out_ptr, out_len)?;
1446			return Ok(());
1447		}
1448
1449		let end = start.saturating_add(out_len as usize).min(input.len());
1450		memory.write(out_ptr, &input[start..end])?;
1451
1452		let bytes_written = (end - start) as u32;
1453		memory.zero(out_ptr.saturating_add(bytes_written), out_len - bytes_written)?;
1454
1455		Ok(())
1456	}

1729	/// Deposit a contract event with the data buffer and optional list of topics.
1730	/// See [pallet_revive_uapi::HostFn::deposit_event]
1731	#[stable]
1732	#[mutating]
1733	fn deposit_event(
1734		&mut self,
1735		memory: &mut M,
1736		topics_ptr: u32,
1737		num_topic: u32,
1738		data_ptr: u32,
1739		data_len: u32,
1740	) -> Result<(), TrapReason> {
1741		self.charge_gas(RuntimeCosts::DepositEvent { num_topic, len: data_len })?;
1742
1743		if num_topic > limits::NUM_EVENT_TOPICS {
1744			return Err(Error::<E::T>::TooManyTopics.into());
1745		}
1746
1747		if data_len > self.ext.max_value_size() {
1748			return Err(Error::<E::T>::ValueTooLarge.into());
1749		}
1750
1751		let topics: Vec<H256> = match num_topic {
1752			0 => Vec::new(),
1753			_ => {
1754				let mut v = Vec::with_capacity(num_topic as usize);
1755				let topics_len = num_topic * H256::len_bytes() as u32;
1756				let buf = memory.read(topics_ptr, topics_len)?;
1757				for chunk in buf.chunks_exact(H256::len_bytes()) {
1758					v.push(H256::from_slice(chunk));
1759				}
1760				v
1761			},
1762		};
1763
1764		let event_data = memory.read(data_ptr, data_len)?;
1765		self.ext.deposit_event(topics, event_data);
1766		Ok(())
1767	}
