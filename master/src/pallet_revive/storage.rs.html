47/// Information for managing an account and its sub trie abstraction.
48/// This is the required info to cache for an account.
49#[derive(Encode, Decode, CloneNoBound, PartialEq, Eq, RuntimeDebug, TypeInfo, MaxEncodedLen)]
50#[scale_info(skip_type_params(T))]
51pub struct ContractInfo<T: Config> {
52	/// Unique ID for the subtree encoded as a bytes vector.
53	pub trie_id: TrieId,
54	/// The code associated with a given account.
55	pub code_hash: sp_core::H256,
56	/// How many bytes of storage are accumulated in this contract's child trie.
57	storage_bytes: u32,
58	/// How many items of storage are accumulated in this contract's child trie.
59	storage_items: u32,
60	/// This records to how much deposit the accumulated `storage_bytes` amount to.
61	pub storage_byte_deposit: BalanceOf<T>,
62	/// This records to how much deposit the accumulated `storage_items` amount to.
63	storage_item_deposit: BalanceOf<T>,
64	/// This records how much deposit is put down in order to pay for the contract itself.
65	///
66	/// We need to store this information separately so it is not used when calculating any refunds
67	/// since the base deposit can only ever be refunded on contract termination.
68	storage_base_deposit: BalanceOf<T>,
69	/// The size of the immutable data of this contract.
70	immutable_data_len: u32,
71}
72
73impl<T: Config> ContractInfo<T> {
74	/// Constructs a new contract info **without** writing it to storage.
75	///
76	/// This returns an `Err` if an contract with the supplied `account` already exists
77	/// in storage.
78	pub fn new(
79		address: &H160,
80		nonce: T::Nonce,
81		code_hash: sp_core::H256,
82	) -> Result<Self, DispatchError> {
83		if <ContractInfoOf<T>>::contains_key(address) {
84			return Err(Error::<T>::DuplicateContract.into());
85		}
86
87		let trie_id = {
88			let buf = ("bcontract_trie_v1", address, nonce).using_encoded(T::Hashing::hash);
89			buf.as_ref()
90				.to_vec()
91				.try_into()
92				.expect("Runtime uses a reasonable hash size. Hence sizeof(T::Hash) <= 128; qed")
93		};
94
95		let contract = Self {
96			trie_id,
97			code_hash,
98			storage_bytes: 0,
99			storage_items: 0,
100			storage_byte_deposit: Zero::zero(),
101			storage_item_deposit: Zero::zero(),
102			storage_base_deposit: Zero::zero(),
103			immutable_data_len: 0,
104		};
105
106		Ok(contract)
107	}

109	/// Associated child trie unique id is built from the hash part of the trie id.
110	pub fn child_trie_info(&self) -> ChildInfo {
111		ChildInfo::new_default(self.trie_id.as_ref())
112	}

114	/// The deposit paying for the accumulated storage generated within the contract's child trie.
115	pub fn extra_deposit(&self) -> BalanceOf<T> {
116		self.storage_byte_deposit.saturating_add(self.storage_item_deposit)
117	}
118
119	/// Same as [`Self::extra_deposit`] but including the base deposit.
120	pub fn total_deposit(&self) -> BalanceOf<T> {
121		self.extra_deposit().saturating_add(self.storage_base_deposit)
122	}

124	/// Returns the storage base deposit of the contract.
125	pub fn storage_base_deposit(&self) -> BalanceOf<T> {
126		self.storage_base_deposit
127	}

129	/// Reads a storage kv pair of a contract.
130	///
131	/// The read is performed from the `trie_id` only. The `address` is not necessary. If the
132	/// contract doesn't store under the given `key` `None` is returned.
133	pub fn read(&self, key: &Key) -> Option<Vec<u8>> {
134		let value = child::get_raw(&self.child_trie_info(), key.hash().as_slice());
135		if_tracing(|t| {
136			t.storage_read(key, value.as_deref());
137		});
138		return value
139	}
140
141	/// Returns `Some(len)` (in bytes) if a storage item exists at `key`.
142	///
143	/// Returns `None` if the `key` wasn't previously set by `set_storage` or
144	/// was deleted.
145	pub fn size(&self, key: &Key) -> Option<u32> {
146		child::len(&self.child_trie_info(), key.hash().as_slice())
147	}

149	/// Update a storage entry into a contract's kv storage.
150	///
151	/// If the `new_value` is `None` then the kv pair is removed. If `take` is true
152	/// a [`WriteOutcome::Taken`] is returned instead of a [`WriteOutcome::Overwritten`].
153	///
154	/// This function also records how much storage was created or removed if a `storage_meter`
155	/// is supplied. It should only be absent for testing or benchmarking code.
156	pub fn write(
157		&self,
158		key: &Key,
159		new_value: Option<Vec<u8>>,
160		storage_meter: Option<&mut meter::NestedMeter<T>>,
161		take: bool,
162	) -> Result<WriteOutcome, DispatchError> {
163		let hashed_key = key.hash();
164		if_tracing(|t| {
165			let old = child::get_raw(&self.child_trie_info(), hashed_key.as_slice());
166			t.storage_write(key, old, new_value.as_deref());
167		});
168
169		self.write_raw(&hashed_key, new_value.as_deref(), storage_meter, take)
170	}
171
172	/// Update a storage entry into a contract's kv storage.
173	/// Function used in benchmarks, which can simulate prefix collision in keys.
174	#[cfg(feature = "runtime-benchmarks")]
175	pub fn bench_write_raw(
176		&self,
177		key: &[u8],
178		new_value: Option<Vec<u8>>,
179		take: bool,
180	) -> Result<WriteOutcome, DispatchError> {
181		self.write_raw(key, new_value.as_deref(), None, take)
182	}

184	fn write_raw(
185		&self,
186		key: &[u8],
187		new_value: Option<&[u8]>,
188		storage_meter: Option<&mut meter::NestedMeter<T>>,
189		take: bool,
190	) -> Result<WriteOutcome, DispatchError> {
191		let child_trie_info = &self.child_trie_info();
192		let (old_len, old_value) = if take {
193			let val = child::get_raw(child_trie_info, key);
194			(val.as_ref().map(|v| v.len() as u32), val)
195		} else {
196			(child::len(child_trie_info, key), None)
197		};
198
199		if let Some(storage_meter) = storage_meter {
200			let mut diff = meter::Diff::default();
201			let key_len = key.len() as u32;
202			match (old_len, new_value.as_ref().map(|v| v.len() as u32)) {
203				(Some(old_len), Some(new_len)) =>
204					if new_len > old_len {
205						diff.bytes_added = new_len - old_len;
206					} else {
207						diff.bytes_removed = old_len - new_len;
208					},
209				(None, Some(new_len)) => {
210					diff.bytes_added = new_len.saturating_add(key_len);
211					diff.items_added = 1;
212				},
213				(Some(old_len), None) => {
214					diff.bytes_removed = old_len.saturating_add(key_len);
215					diff.items_removed = 1;
216				},
217				(None, None) => (),
218			}
219			storage_meter.charge(&diff);
220		}
221
222		match &new_value {
223			Some(new_value) => child::put_raw(child_trie_info, key, new_value),
224			None => child::kill(child_trie_info, key),
225		}
226
227		Ok(match (old_len, old_value) {
228			(None, _) => WriteOutcome::New,
229			(Some(old_len), None) => WriteOutcome::Overwritten(old_len),
230			(Some(_), Some(old_value)) => WriteOutcome::Taken(old_value),
231		})
232	}

234	/// Sets and returns the contract base deposit.
235	///
236	/// The base deposit is updated when the `code_hash` of the contract changes, as it depends on
237	/// the deposit paid to upload the contract's code. It also depends on the size of immutable
238	/// storage which is also changed when the code hash of a contract is changed.
239	pub fn update_base_deposit(&mut self, code_deposit: BalanceOf<T>) -> BalanceOf<T> {
240		let contract_deposit = Diff {
241			bytes_added: (self.encoded_size() as u32).saturating_add(self.immutable_data_len),
242			items_added: if self.immutable_data_len == 0 { 1 } else { 2 },
243			..Default::default()
244		}
245		.update_contract::<T>(None)
246		.charge_or_zero();
247
248		// Instantiating the contract prevents its code to be deleted, therefore the base deposit
249		// includes a fraction (`T::CodeHashLockupDepositPercent`) of the original storage deposit
250		// to prevent abuse.
251		let code_deposit = T::CodeHashLockupDepositPercent::get().mul_ceil(code_deposit);
252
253		let deposit = contract_deposit.saturating_add(code_deposit);
254		self.storage_base_deposit = deposit;
255		deposit
256	}

265	/// Calculates the weight that is necessary to remove one key from the trie and how many
266	/// of those keys can be deleted from the deletion queue given the supplied weight limit.
267	pub fn deletion_budget(meter: &WeightMeter) -> (Weight, u32) {
268		let base_weight = T::WeightInfo::on_process_deletion_queue_batch();
269		let weight_per_key = T::WeightInfo::on_initialize_per_trie_key(1) -
270			T::WeightInfo::on_initialize_per_trie_key(0);
271
272		// `weight_per_key` being zero makes no sense and would constitute a failure to
273		// benchmark properly. We opt for not removing any keys at all in this case.
274		let key_budget = meter
275			.limit()
276			.saturating_sub(base_weight)
277			.checked_div_per_component(&weight_per_key)
278			.unwrap_or(0) as u32;
279
280		(weight_per_key, key_budget)
281	}
282
283	/// Delete as many items from the deletion queue possible within the supplied weight limit.
284	pub fn process_deletion_queue_batch(meter: &mut WeightMeter) {
285		if meter.try_consume(T::WeightInfo::on_process_deletion_queue_batch()).is_err() {
286			return
287		};
288
289		let mut queue = <DeletionQueueManager<T>>::load();
290		if queue.is_empty() {
291			return;
292		}
293
294		let (weight_per_key, budget) = Self::deletion_budget(&meter);
295		let mut remaining_key_budget = budget;
296		while remaining_key_budget > 0 {
297			let Some(entry) = queue.next() else { break };
298
299			#[allow(deprecated)]
300			let outcome = child::kill_storage(
301				&ChildInfo::new_default(&entry.trie_id),
302				Some(remaining_key_budget),
303			);
304
305			match outcome {
306				// This happens when our budget wasn't large enough to remove all keys.
307				KillStorageResult::SomeRemaining(keys_removed) => {
308					remaining_key_budget.saturating_reduce(keys_removed);
309					break
310				},
311				KillStorageResult::AllRemoved(keys_removed) => {
312					entry.remove();
313					// charge at least one key even if none were removed.
314					remaining_key_budget = remaining_key_budget.saturating_sub(keys_removed.max(1));
315				},
316			};
317		}
318
319		meter.consume(weight_per_key.saturating_mul(u64::from(budget - remaining_key_budget)))
320	}

338/// Information about what happened to the pre-existing value when calling [`ContractInfo::write`].
339#[cfg_attr(any(test, feature = "runtime-benchmarks"), derive(Debug, PartialEq))]
340pub enum WriteOutcome {
341	/// No value existed at the specified key.
342	New,
343	/// A value of the returned length was overwritten.
344	Overwritten(u32),
345	/// The returned value was taken out of storage before being overwritten.
346	///
347	/// This is only returned when specifically requested because it causes additional work
348	/// depending on the size of the pre-existing value. When not requested [`Self::Overwritten`]
349	/// is returned instead.
350	Taken(Vec<u8>),
351}
352
353impl WriteOutcome {
354	/// Extracts the size of the overwritten value or `0` if there
355	/// was no value in storage.
356	pub fn old_len(&self) -> u32 {
357		match self {
358			Self::New => 0,
359			Self::Overwritten(len) => *len,
360			Self::Taken(value) => value.len() as u32,
361		}
362	}
363
364	/// Extracts the size of the overwritten value or `SENTINEL` if there
365	/// was no value in storage.
366	///
367	/// # Note
368	///
369	/// We cannot use `0` as sentinel value because there could be a zero sized
370	/// storage entry which is different from a non existing one.
371	pub fn old_len_with_sentinel(&self) -> u32 {
372		match self {
373			Self::New => SENTINEL,
374			Self::Overwritten(len) => *len,
375			Self::Taken(value) => value.len() as u32,
376		}
377	}
378}

380/// Manage the removal of contracts storage that are marked for deletion.
381///
382/// When a contract is deleted by calling `seal_terminate` it becomes inaccessible
383/// immediately, but the deletion of the storage items it has accumulated is performed
384/// later by pulling the contract from the queue in the `on_idle` hook.
385#[derive(Encode, Decode, TypeInfo, MaxEncodedLen, DefaultNoBound, Clone)]
386#[scale_info(skip_type_params(T))]
387pub struct DeletionQueueManager<T: Config> {
388	/// Counter used as a key for inserting a new deleted contract in the queue.
389	/// The counter is incremented after each insertion.
390	insert_counter: u32,
391	/// The index used to read the next element to be deleted in the queue.
392	/// The counter is incremented after each deletion.
393	delete_counter: u32,
394
395	_phantom: PhantomData<T>,
396}
397
398/// View on a contract that is marked for deletion.
399struct DeletionQueueEntry<'a, T: Config> {
400	/// the trie id of the contract to delete.
401	trie_id: TrieId,
402
403	/// A mutable reference on the queue so that the contract can be removed, and none can be added
404	/// or read in the meantime.
405	queue: &'a mut DeletionQueueManager<T>,
406}
407
408impl<'a, T: Config> DeletionQueueEntry<'a, T> {
409	/// Remove the contract from the deletion queue.
410	fn remove(self) {
411		<DeletionQueue<T>>::remove(self.queue.delete_counter);
412		self.queue.delete_counter = self.queue.delete_counter.wrapping_add(1);
413		<DeletionQueueCounter<T>>::set(self.queue.clone());
414	}
415}
416
417impl<T: Config> DeletionQueueManager<T> {
418	/// Load the `DeletionQueueCounter`, so we can perform read or write operations on the
419	/// DeletionQueue storage.
420	fn load() -> Self {
421		<DeletionQueueCounter<T>>::get()
422	}
423
424	/// Returns `true` if the queue contains no elements.
425	fn is_empty(&self) -> bool {
426		self.insert_counter.wrapping_sub(self.delete_counter) == 0
427	}
428
429	/// Insert a contract in the deletion queue.
430	fn insert(&mut self, trie_id: TrieId) {
431		<DeletionQueue<T>>::insert(self.insert_counter, trie_id);
432		self.insert_counter = self.insert_counter.wrapping_add(1);
433		<DeletionQueueCounter<T>>::set(self.clone());
434	}
435
436	/// Fetch the next contract to be deleted.
437	///
438	/// Note:
439	/// we use the delete counter to get the next value to read from the queue and thus don't pay
440	/// the cost of an extra call to `sp_io::storage::next_key` to lookup the next entry in the map
441	fn next(&mut self) -> Option<DeletionQueueEntry<T>> {
442		if self.is_empty() {
443			return None
444		}
445
446		let entry = <DeletionQueue<T>>::get(self.delete_counter);
447		entry.map(|trie_id| DeletionQueueEntry { trie_id, queue: self })
448	}
449}
