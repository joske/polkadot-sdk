24impl TransactionUnsigned {
25	/// Return the bytes to be signed by the private key.
26	pub fn unsigned_payload(&self) -> Vec<u8> {
27		use TransactionUnsigned::*;
28		let mut s = rlp::RlpStream::new();
29		match self {
30			Transaction2930Unsigned(ref tx) => {
31				s.append(&tx.r#type.value());
32				s.append(tx);
33			},
34			Transaction1559Unsigned(ref tx) => {
35				s.append(&tx.r#type.value());
36				s.append(tx);
37			},
38			Transaction4844Unsigned(ref tx) => {
39				s.append(&tx.r#type.value());
40				s.append(tx);
41			},
42			TransactionLegacyUnsigned(ref tx) => {
43				s.append(tx);
44			},
45		}
46
47		s.out().to_vec()
48	}
49}

51impl TransactionSigned {
52	/// Extract the unsigned transaction from a signed transaction.
53	pub fn unsigned(self) -> TransactionUnsigned {
54		use TransactionSigned::*;
55		use TransactionUnsigned::*;
56		match self {
57			Transaction2930Signed(tx) => Transaction2930Unsigned(tx.transaction_2930_unsigned),
58			Transaction1559Signed(tx) => Transaction1559Unsigned(tx.transaction_1559_unsigned),
59			Transaction4844Signed(tx) => Transaction4844Unsigned(tx.transaction_4844_unsigned),
60			TransactionLegacySigned(tx) =>
61				TransactionLegacyUnsigned(tx.transaction_legacy_unsigned),
62		}
63	}
64
65	/// Encode the Ethereum transaction into bytes.
66	pub fn signed_payload(&self) -> Vec<u8> {
67		use TransactionSigned::*;
68		let mut s = rlp::RlpStream::new();
69		match self {
70			Transaction2930Signed(ref tx) => {
71				s.append(&tx.transaction_2930_unsigned.r#type.value());
72				s.append(tx);
73			},
74			Transaction1559Signed(ref tx) => {
75				s.append(&tx.transaction_1559_unsigned.r#type.value());
76				s.append(tx);
77			},
78			Transaction4844Signed(ref tx) => {
79				s.append(&tx.transaction_4844_unsigned.r#type.value());
80				s.append(tx);
81			},
82			TransactionLegacySigned(ref tx) => {
83				s.append(tx);
84			},
85		}
86
87		s.out().to_vec()
88	}
89
90	/// Decode the Ethereum transaction from bytes.
91	pub fn decode(data: &[u8]) -> Result<Self, rlp::DecoderError> {
92		if data.len() < 1 {
93			return Err(rlp::DecoderError::RlpIsTooShort);
94		}
95		match data[0] {
96			TYPE_EIP2930 => rlp::decode::<Transaction2930Signed>(&data[1..]).map(Into::into),
97			TYPE_EIP1559 => rlp::decode::<Transaction1559Signed>(&data[1..]).map(Into::into),
98			TYPE_EIP4844 => rlp::decode::<Transaction4844Signed>(&data[1..]).map(Into::into),
99			_ => rlp::decode::<TransactionLegacySigned>(data).map(Into::into),
100		}
101	}
102}

104impl TransactionUnsigned {
105	/// Get a signed transaction payload with a dummy 65 bytes signature.
106	pub fn dummy_signed_payload(self) -> Vec<u8> {
107		const DUMMY_SIGNATURE: [u8; 65] = [1u8; 65];
108		self.with_signature(DUMMY_SIGNATURE).signed_payload()
109	}

112/// See <https://eips.ethereum.org/EIPS/eip-155>
113impl Encodable for TransactionLegacyUnsigned {
114	fn rlp_append(&self, s: &mut rlp::RlpStream) {
115		if let Some(chain_id) = self.chain_id {
116			s.begin_list(9);
117			s.append(&self.nonce);
118			s.append(&self.gas_price);
119			s.append(&self.gas);
120			match self.to {
121				Some(ref to) => s.append(to),
122				None => s.append_empty_data(),
123			};
124			s.append(&self.value);
125			s.append(&self.input.0);
126			s.append(&chain_id);
127			s.append(&0u8);
128			s.append(&0u8);
129		} else {
130			s.begin_list(6);
131			s.append(&self.nonce);
132			s.append(&self.gas_price);
133			s.append(&self.gas);
134			match self.to {
135				Some(ref to) => s.append(to),
136				None => s.append_empty_data(),
137			};
138			s.append(&self.value);
139			s.append(&self.input.0);
140		}
141	}
142}

144impl Decodable for TransactionLegacyUnsigned {
145	fn decode(rlp: &rlp::Rlp) -> Result<Self, rlp::DecoderError> {
146		Ok(TransactionLegacyUnsigned {
147			nonce: rlp.val_at(0)?,
148			gas_price: rlp.val_at(1)?,
149			gas: rlp.val_at(2)?,
150			to: {
151				let to = rlp.at(3)?;
152				if to.is_empty() {
153					None
154				} else {
155					Some(to.as_val()?)
156				}
157			},
158			value: rlp.val_at(4)?,
159			input: Bytes(rlp.val_at(5)?),
160			chain_id: rlp.val_at(6).ok(),
161			..Default::default()
162		})
163	}

413/// See <https://eips.ethereum.org/EIPS/eip-155>
414impl Decodable for TransactionLegacySigned {
415	fn decode(rlp: &rlp::Rlp) -> Result<Self, rlp::DecoderError> {
416		let v: U256 = rlp.val_at(6)?;
417
418		let extract_chain_id = |v: U256| {
419			if v.ge(&35u32.into()) {
420				Some((v - 35) / 2)
421			} else {
422				None
423			}
424		};
425
426		Ok(TransactionLegacySigned {
427			transaction_legacy_unsigned: {
428				TransactionLegacyUnsigned {
429					nonce: rlp.val_at(0)?,
430					gas_price: rlp.val_at(1)?,
431					gas: rlp.val_at(2)?,
432					to: {
433						let to = rlp.at(3)?;
434						if to.is_empty() {
435							None
436						} else {
437							Some(to.as_val()?)
438						}
439					},
440					value: rlp.val_at(4)?,
441					input: Bytes(rlp.val_at(5)?),
442					chain_id: extract_chain_id(v).map(|v| v.into()),
443					r#type: TypeLegacy {},
444				}
445			},
446			v,
447			r: rlp.val_at(7)?,
448			s: rlp.val_at(8)?,
449		})
450	}
451}
